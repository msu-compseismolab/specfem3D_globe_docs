33 ! interpolates model value for given location xi/eta/gamma within element ielem using GLL basis functions
34

36 
37  implicit none
38 
39  double precision,intent(in):: xi,eta,gamma
40  integer,intent(in):: ielem

45  real(kind=CUSTOM_REAL),intent(out) :: val

47  ! Gauss-Lobatto-Legendre points of integration and weights

52  ! local parameters
53  double precision :: hxir(NGLLX), hpxir(NGLLX), hetar(NGLLY), hpetar(NGLLY), &
54  hgammar(NGLLZ), hpgammar(NGLLZ)

78  subroutine interpolate_limited(xi,eta,gamma,ielem, &
79  nspec,model, &
80  val,xigll,yigll,zigll, &
81  i_selected,j_selected,k_selected)
82 
83 ! interpolates model value for given location xi/eta/gamma within element ielem using GLL basis function;
84 ! checks interpolated value with closest point value and limits value range to 1 % (PERCENTAGE_LIMIT)
85 
86  use constants, only: custom_real,ngllx,nglly,ngllz
87 
88  implicit none
89 
90  double precision,intent(in):: xi,eta,gamma
91  integer,intent(in):: ielem
92 
93  integer,intent(in):: nspec
94  real(kind=CUSTOM_REAL),dimension(NGLLX,NGLLY,NGLLZ,nspec),intent(in) :: model
95 
96  real(kind=CUSTOM_REAL),intent(out) :: val
97 
98  ! Gauss-Lobatto-Legendre points of integration and weights
99  double precision, dimension(NGLLX),intent(in) :: xigll
100  double precision, dimension(NGLLY),intent(in) :: yigll
101  double precision, dimension(NGLLZ),intent(in) :: zigll
102 
103  integer,intent(in):: i_selected,j_selected,k_selected
104 
105  ! local parameters
106  double precision :: hxir(NGLLX), hpxir(NGLLX), hetar(NGLLY), hpetar(NGLLY), &
107  hgammar(NGLLZ), hpgammar(NGLLZ)
108  integer:: i,j,k
109  real(kind=CUSTOM_REAL) :: val_initial,val_avg,pert,pert_limit
110 
111  !--------------------------------------------------------------------
112 
113  ! percentage
114  real(kind=CUSTOM_REAL), parameter :: PERCENTAGE_LIMIT = 0.01
115 
116  !--------------------------------------------------------------------
117 
118  ! interpolation weights
119  call lagrange_any(xi,ngllx,xigll,hxir,hpxir)
120  call lagrange_any(eta,nglly,yigll,hetar,hpetar)
121  call lagrange_any(gamma,ngllz,zigll,hgammar,hpgammar)
122 
123  ! interpolates value
124  val = 0.0_custom_real
125  do k = 1, ngllz
126  do j = 1, nglly
127  do i = 1, ngllx
128  val = val + hxir(i) * hetar(j) * hgammar(k) * model(i,j,k,ielem)
129  enddo
130  enddo
131  enddo
132 
133  ! note: interpolation of values close to the surface or 3D moho encounters problems;
134  ! this is a fall-back to the closest point value
135  !
136  ! uses average/closest point value if too far off
137 
138  ! closest point value
139  val_initial = model(i_selected,j_selected,k_selected,ielem)
140 
141  ! average value
142  val_avg = sum(model(:,:,:,ielem)) / ngllx / nglly / ngllz
143 
144  ! initial difference
145  pert = abs(val_initial - val_avg)
146 
147  ! upper/lower perturbation bound
148  pert_limit = percentage_limit * abs(val_avg)
149  if (pert > pert_limit) pert_limit = pert
150 
151  ! within a certain percentage range
152  if (abs(val - val_avg ) > pert_limit) then
153  val = val_initial
154  endif
155 
156  end subroutine interpolate_limited
