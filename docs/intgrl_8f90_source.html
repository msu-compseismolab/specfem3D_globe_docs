28  subroutine intgrl(sumval,r,nir,ner,f,s1,s2,s3)
29 
30 ! Computes the integral of f[i]*r[i]*r[i] from i=nir to i=ner for
31 ! radii values as in model PREM_an640
32 
33  implicit none
34 
35 ! Argument variables
36  integer :: ner,nir
37  double precision :: f(640),r(640),s1(640),s2(640)
38  double precision :: s3(640),sumval
39 
40 ! Local variables
41  double precision, parameter :: third = 1.0d0/3.0d0
42  double precision, parameter :: fifth = 1.0d0/5.0d0
43  double precision, parameter :: sixth = 1.0d0/6.0d0
44 
45  double precision :: rji,yprime(640)
46  double precision :: s1l,s2l,s3l
47 
48  integer :: i,j,n,kdis(28)
49  integer :: ndis,nir1
50 
51  data kdis/163,323,336,517,530,540,565,590,609,619,626,633,16*0/
52 
53  ndis = 12
54  n = 640
55 
56  call deriv(f,yprime,n,r,ndis,kdis,s1,s2,s3)
57 
58  nir1 = nir + 1
59  sumval = 0.0d0
60  do i=nir1,ner
61  j = i-1
62  rji = r(i) - r(j)
63  s1l = s1(j)
64  s2l = s2(j)
65  s3l = s3(j)
66  sumval = sumval + r(j)*r(j)*rji*(f(j) &
67  + rji*(0.5d0*s1l + rji*(third*s2l + rji*0.25d0*s3l))) &
68  + 2.0d0*r(j)*rji*rji*(0.5d0*f(j) + rji*(third*s1l + rji*(0.25d0*s2l + rji*fifth*s3l))) &
69  + rji*rji*rji*(third*f(j) + rji*(0.25d0*s1l + rji*(fifth*s2l + rji*sixth*s3l)))
70  enddo
71 
72  end subroutine intgrl

76  subroutine deriv(y,yprime,n,r,ndis,kdis,s1,s2,s3)
77 
78  implicit none
79 
80 ! Argument variables
81  integer :: kdis(28),n,ndis
82  double precision :: r(n),s1(n),s2(n),s3(n)
83  double precision :: y(n),yprime(n)
84 
85 ! Local variables
86  integer :: i,j,j1,j2
87  integer :: k,nd,ndp
88  double precision :: a0,b0,b1
89  double precision :: f(3,1000),h,h2,h2a
90  double precision :: h2b,h3a,ha,s13
91  double precision :: s21,s32,yy(3)
92 
93  yy(1) = 0.d0
94  yy(2) = 0.d0
95  yy(3) = 0.d0
96 
97  ndp=ndis+1
98  do 3 nd = 1,ndp
99  if (nd == 1) goto 4
100  if (nd == ndp) goto 5
101  j1=kdis(nd-1)+1
102  j2=kdis(nd)-2
103  goto 6
104  4 j1 = 1
105  j2=kdis(1)-2
106  goto 6
107  5 j1=kdis(ndis)+1
108  j2=n-2
109  6 if ((j2+1-j1) > 0) goto 11
110  j2=j2+2
111  yy(1)=(y(j2)-y(j1))/(r(j2)-r(j1))
112  s1(j1)=yy(1)
113  s1(j2)=yy(1)
114  s2(j1)=yy(2)
115  s2(j2)=yy(2)
116  s3(j1)=yy(3)
117  s3(j2)=yy(3)
118  goto 3
119  11 a0=0.0d0
120  if (j1 == 1) goto 7
121  h=r(j1+1)-r(j1)
122  h2=r(j1+2)-r(j1)
123  yy(1)=h*h2*(h2-h)
124  h=h*h
125  h2=h2*h2
126  b0=(y(j1)*(h-h2)+y(j1+1)*h2-y(j1+2)*h)/yy(1)
127  goto 8
128  7 b0=0.0d0
129  8 b1=b0
130 
131  if (j2 > 1000) stop 'Error in subroutine deriv for j2'
132 
133  do i=j1,j2
134  h=r(i+1)-r(i)
135  yy(1)=y(i+1)-y(i)
136  h2=h*h
137  ha=h-a0
138  h2a=h-2.0d0*a0
139  h3a=2.0d0*h-3.0d0*a0
140  h2b=h2*b0
141  s1(i)=h2/ha
142  s2(i)=-ha/(h2a*h2)
143  s3(i)=-h*h2a/h3a
144  f(1,i)=(yy(1)-h*b0)/(h*ha)
145  f(2,i)=(h2b-yy(1)*(2.0d0*h-a0))/(h*h2*h2a)
146  f(3,i)=-(h2b-3.0d0*yy(1)*ha)/(h*h3a)
147  a0=s3(i)
148  b0=f(3,i)
149  enddo
150 
151  i=j2+1
152  h=r(i+1)-r(i)
153  yy(1)=y(i+1)-y(i)
154  h2=h*h
155  ha=h-a0
156  h2a=h*ha
157  h2b=h2*b0-yy(1)*(2.d0*h-a0)
158  s1(i)=h2/ha
159  f(1,i)=(yy(1)-h*b0)/h2a
160  ha=r(j2)-r(i+1)
161  yy(1)=-h*ha*(ha+h)
162  ha=ha*ha
163  yy(1)=(y(i+1)*(h2-ha)+y(i)*ha-y(j2)*h2)/yy(1)
164  s3(i)=(yy(1)*h2a+h2b)/(h*h2*(h-2.0d0*a0))
165  s13=s1(i)*s3(i)
166  s2(i)=f(1,i)-s13
167 
168  do j=j1,j2
169  k=i-1
170  s32=s3(k)*s2(i)
171  s1(i)=f(3,k)-s32
172  s21=s2(k)*s1(i)
173  s3(k)=f(2,k)-s21
174  s13=s1(k)*s3(k)
175  s2(k)=f(1,k)-s13
176  i=k
177  enddo
178 
179  s1(i)=b1
180  j2=j2+2
181  s1(j2)=yy(1)
182  s2(j2)=yy(2)
183  s3(j2)=yy(3)
184  3 continue
185 
186  do i = 1,n
187  yprime(i)=s1(i)
188  enddo
189 
190  end subroutine deriv
