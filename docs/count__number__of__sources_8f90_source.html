28  subroutine count_number_of_sources(NSOURCES)
29 
30 ! count the total number of sources in the CMTSOLUTION file
31 ! there are NLINES_PER_CMTSOLUTION_SOURCE lines per source in that file
32 
33  use constants
34  use shared_parameters, only: number_of_simultaneous_runs,use_force_point_source
35 
36  implicit none
37 
38  integer, intent(out) :: NSOURCES
39 
40  integer :: ios,icounter,nline
41 
42  character(len=MAX_STRING_LEN) :: SOURCE_FILE, path_to_add
43  character(len=MAX_STRING_LEN) :: dummystring
44 
45  if (use_force_point_source) then
46  ! FORCESOLUTION file
47  source_file = 'DATA/FORCESOLUTION'
48  nline=nlines_per_forcesolution_source
49  else
50  ! CMTSOLUTION file
51  source_file = 'DATA/CMTSOLUTION'
52  nline=nlines_per_cmtsolution_source
53  endif
54 
55  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
56  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
57  source_file = path_to_add(1:len_trim(path_to_add))//source_file(1:len_trim(source_file))
58  endif
59 
60  open(unit=iin,file=trim(source_file),status='old',action='read',iostat=ios)
61  if (ios /= 0) then
62  print *,'Error opening CMTSOLUTION file: ',trim(source_file)
63  stop 'Error opening CMTSOLUTION file'
64  endif
65 
66  icounter = 0
67  do while(ios == 0)
68  read(iin,"(a)",iostat=ios) dummystring
69  if (ios == 0) icounter = icounter + 1
70  enddo
71  close(iin)
72 
73  if (mod(icounter,nline) /= 0) then
74  if (use_force_point_source) then
75  stop 'total number of lines in FORCESOLUTION file should be a multiple of NLINES_PER_FORCESOLUTION_SOURCE'
76  else
77  stop 'total number of lines in CMTSOLUTION file should be a multiple of NLINES_PER_CMTSOLUTION_SOURCE'
78  endif
79  endif
80 
81  nsources = icounter / nline
82 
83  if (nsources < 1) stop 'need at least one source in CMTSOLUTION or FORCESOLUTION file'
84 
85  end subroutine count_number_of_sources
