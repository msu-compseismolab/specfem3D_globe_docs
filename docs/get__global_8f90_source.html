70  subroutine get_global_indirect_addressing(nspec,nglob,ibool)
71 
72 !
73 !- we can create a new indirect addressing to reduce cache misses
74 ! (put into this subroutine but compiler keeps on complaining that it can't vectorize loops...)
75 
76  use constants
77 
78  implicit none
79 
80  integer,intent(in) :: nspec,nglob
81  integer, dimension(NGLLX,NGLLY,NGLLZ,nspec),intent(inout) :: ibool
82 
83  ! local parameters
84  ! mask to sort ibool
85  integer, dimension(:), allocatable :: mask_ibool
86  integer, dimension(:,:,:,:), allocatable :: copy_ibool_ori
87  integer :: inumber
88  integer:: i,j,k,ispec,ier
89 
90  ! copies original array
91  allocate(copy_ibool_ori(ngllx,nglly,ngllz,nspec), &
92  mask_ibool(nglob), &
93  stat=ier)
94  if (ier /= 0) stop 'Error allocating local arrays in get_global_indirect_addressing'
95 
96  ! initializes arrays
97  mask_ibool(:) = -1
98  copy_ibool_ori(:,:,:,:) = ibool(:,:,:,:)
99 
100  ! reduces misses
101  inumber = 0
102  do ispec = 1,nspec
103  do k = 1,ngllz
104  do j = 1,nglly
105  do i = 1,ngllx
106  if (mask_ibool(copy_ibool_ori(i,j,k,ispec)) == -1) then
107  ! creates a new point
108  inumber = inumber + 1
109  ibool(i,j,k,ispec) = inumber
110  mask_ibool(copy_ibool_ori(i,j,k,ispec)) = inumber
111  else
112  ! uses an existing point created previously
113  ibool(i,j,k,ispec) = mask_ibool(copy_ibool_ori(i,j,k,ispec))
114  endif
115  enddo
116  enddo
117  enddo
118  enddo
119 
120  ! cleanup
121  deallocate(copy_ibool_ori)
122  deallocate(mask_ibool)
123 
124  end subroutine get_global_indirect_addressing
