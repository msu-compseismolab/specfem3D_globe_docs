37  You'll notice that the heart of the parser is a complex regular
38  expression that is compiled within the C code, and then used to split
39  the lines appropriately. It does all the heavy lifting. I don't know of
40  any way to do this in Fortran. I believe that to accomplish this in
41  Fortran, you'd have to write a lot of procedural string manipulation
42  code, for which Fortran is not very well suited.
43 
44  But Fortran-C mixes are pretty common these days, so I would not expect
45  any problems on that account. There are no wrapper functions used: just
46  the C routine called directly from a Fortran routine. Also, regarding
47  the use of C, I assumed this would not be a problem since there are
48  already a few C files that make up part of the build.

118 void
119 FC_FUNC_(param_read,PARAM_READ)(char * string_read, int * string_read_len, char * name, int * name_len, int * ierr)
120 {
121  char * namecopy;
122  char * blank;
123  int status;
124  regex_t compiled_pattern;
125  char line[LINE_MAX];
126  int regret;
127  regmatch_t parameter[3];
128  char * keyword;
129  char * value;
130  size_t value_len;
131 
132  // Trim the keyword name we're looking for.
133  namecopy = strndup(name, *name_len);
134  blank = strchr(namecopy, ' ');
135  if (blank != NULL) {
136  namecopy[blank - namecopy] = '\0';
137  }
138  /* Regular expression for parsing lines from param file.
139  ** Good luck reading this regular expression. Basically, the lines of
140  ** the parameter file should be of the form 'parameter = value',
141  ** optionally followed by a #-delimited comment.
142  ** 'value' can be any number of space- or tab-separated words. Blank
143  ** lines, lines containing only white space and lines whose first non-
144  ** whitespace character is '#' are ignored. White space is generally
145  ** ignored. As you will see later in the code, if both parameter and
146  ** value are not specified the line is ignored.
147  */
148  char pattern[] = "^[ \t]*([^# \t]+)[ \t]*=[ \t]*([^# \t]+([ \t]+[^# \t]+)*)";
149 
150  // Compile the regular expression.
151  status = regcomp(&compiled_pattern, pattern, REG_EXTENDED);
152  if (status != 0) {
153  printf("regcomp returned error %d\n", status);
154  }
155  // Position the open file to the beginning.
156  if (fseek(fid, 0, SEEK_SET) != 0) {
157  printf("Can't seek to beginning of parameter file\n");
158  *ierr = 1;
159  regfree(&compiled_pattern);
160  return;
161  }
162  // Read every line in the file.
163  while (fgets(line, LINE_MAX, fid) != NULL) {
164  // Get rid of the ending newline.
165  int linelen = strlen(line);
166  if (line[linelen-1] == '\n') {
167  line[linelen-1] = '\0';
168  }
169  /* Test if line matches the regular expression pattern, if so
170  ** return position of keyword and value */
171  regret = regexec(&compiled_pattern, line, 3, parameter, 0);
172  // If no match, check the next line.
173  if (regret == REG_NOMATCH) {
174  continue;
175  }
176  // If any error, bail out with an error message.
177  if (regret != 0) {
178  printf("regexec returned error %d\n", regret);
179  *ierr = 1;
180  regfree(&compiled_pattern);
181  return;
182  }
183  // printf("Line read = %s\n", line);
184  // If we have a match, extract the keyword from the line.
185  keyword = strndup(line+parameter[1].rm_so, parameter[1].rm_eo-parameter[1].rm_so);
186  // If the keyword is not the one we're looking for, check the next line.
187  if (strcmp(keyword, namecopy) != 0) {
188  free(keyword);
189  continue;
190  }
191  free(keyword);
192  regfree(&compiled_pattern);
193  // If it matches, extract the value from the line.
194  value = strndup(line+parameter[2].rm_so, parameter[2].rm_eo-parameter[2].rm_so);
195  // Clear out the return string with blanks, copy the value into it, and return.
196  memset(string_read, ' ', *string_read_len);
197  value_len = strlen(value);
198  if (value_len > (size_t)*string_read_len)
199  value_len = *string_read_len;
200  strncpy(string_read, value, value_len);
201  free(value);
202  free(namecopy);
203  *ierr = 0;
204  return;
205  }
206  // If no keyword matches, print out error and die.
207  printf("No match in parameter file for keyword '%s'\n", namecopy);
208  free(namecopy);
209  regfree(&compiled_pattern);
210  *ierr = 1;
211  return;
212 }
