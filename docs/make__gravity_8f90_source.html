28  subroutine make_gravity(nspl,rspl,gspl,gspl2,ONE_CRUST)
29 
30 ! creates a spline for the gravity profile in PREM
31 ! radius and density are non-dimensional
32 
33  use constants
34 
35  implicit none
36 
37  integer nspl
38 
39  logical ONE_CRUST
40 
41 ! radius of the Earth for gravity calculation
42  double precision, parameter :: R_EARTH_GRAVITY = 6371000.d0
43 ! radius of the ocean floor for gravity calculation
44  double precision, parameter :: ROCEAN_GRAVITY = 6368000.d0
45 
46  double precision rspl(NR),gspl(NR),gspl2(NR)
47 
48  integer i
49  double precision ROCEAN,RMIDDLE_CRUST,RMOHO,R80,R220,R400,R600,R670, &
50  R771,RTOPDDOUBLEPRIME,RCMB,RICB
51  double precision r_icb,r_cmb,r_topddoubleprime,r_771,r_670,r_600
52  double precision r_400,r_220,r_80,r_moho,r_middle_crust,r_ocean,r_0
53  double precision r(NR),rho(NR),g(NR),i_rho
54  double precision s1(NR),s2(NR),s3(NR)
55  double precision yp1,ypn
56 
57 ! PREM
58  rocean = rocean_gravity ! PREM defines this as 6368000.d0
59  rmiddle_crust = 6356000.d0
60  rmoho = 6346600.d0 ! PREM moho depth at 24.4 km
61  r80 = 6291000.d0
62  r220 = 6151000.d0
63  r400 = 5971000.d0
64  r600 = 5771000.d0
65  r670 = 5701000.d0
66  r771 = 5600000.d0
67  rtopddoubleprime = 3630000.d0
68  rcmb = 3480000.d0
69  ricb = 1221000.d0
70 
71 ! non-dimensionalize
72  r_icb = ricb/r_earth_gravity
73  r_cmb = rcmb/r_earth_gravity
74  r_topddoubleprime = rtopddoubleprime/r_earth_gravity
75  r_771 = r771/r_earth_gravity
76  r_670 = r670/r_earth_gravity
77  r_600 = r600/r_earth_gravity
78  r_400 = r400/r_earth_gravity
79  r_220 = r220/r_earth_gravity
80  r_80 = r80/r_earth_gravity
81  r_moho = rmoho/r_earth_gravity
82  r_middle_crust = rmiddle_crust/r_earth_gravity
83  r_ocean = rocean/r_earth_gravity
84  r_0 = 1.d0
85 
86  do i=1,163
87  r(i) = r_icb*dble(i-1)/dble(162)
88  enddo
89  do i=164,323
90  r(i) = r_icb+(r_cmb-r_icb)*dble(i-164)/dble(159)
91  enddo
92  do i=324,336
93  r(i) = r_cmb+(r_topddoubleprime-r_cmb)*dble(i-324)/dble(12)
94  enddo
95  do i=337,517
96  r(i) = r_topddoubleprime+(r_771-r_topddoubleprime)*dble(i-337)/dble(180)
97  enddo
98  do i=518,530
99  r(i) = r_771+(r_670-r_771)*dble(i-518)/dble(12)
100  enddo
101  do i=531,540
102  r(i) = r_670+(r_600-r_670)*dble(i-531)/dble(9)
103  enddo
104  do i=541,565
105  r(i) = r_600+(r_400-r_600)*dble(i-541)/dble(24)
106  enddo
107  do i=566,590
108  r(i) = r_400+(r_220-r_400)*dble(i-566)/dble(24)
109  enddo
110  do i=591,609
111  r(i) = r_220+(r_80-r_220)*dble(i-591)/dble(18)
112  enddo
113  do i=610,619
114  r(i) = r_80+(r_moho-r_80)*dble(i-610)/dble(9)
115  enddo
116  do i=620,626
117  r(i) = r_moho+(r_middle_crust-r_moho)*dble(i-620)/dble(6)
118  enddo
119  do i=627,633
120  r(i) = r_middle_crust+(r_ocean-r_middle_crust)*dble(i-627)/dble(6)
121  enddo
122  do i=634,nr
123  r(i) = r_ocean+(r_0-r_ocean)*dble(i-634)/dble(6)
124  enddo
125 
126 ! use PREM to get the density profile for ellipticity (fine for other 1D reference models)
127  do i = 1,nr
128  call prem_density(r(i),rho(i),one_crust,ricb,rcmb,rtopddoubleprime, &
129  r600,r670,r220,r771,r400,r80,rmoho,rmiddle_crust,rocean)
130  enddo
131 
132  g(1)=0.0d0
133  do i=2,nr
134  call intgrl(i_rho,r,1,i,rho,s1,s2,s3)
135  g(i)=4.0d0*i_rho/(r(i)*r(i))
136  enddo
137 
138 !
139 ! get ready to spline g
140 !
141  nspl = 1
142  rspl(1)=r(1)
143  gspl(1)=g(1)
144  do i=2,nr
145  if (r(i) /= r(i-1)) then
146  nspl=nspl+1
147  rspl(nspl)=r(i)
148  gspl(nspl)=g(i)
149  endif
150  enddo
151  yp1=(4.0d0/3.0d0)*rho(1)
152  ypn=4.0d0*rho(nr)-2.0d0*g(nr)/r(nr)
153  call spline_construction(rspl,gspl,nspl,yp1,ypn,gspl2)
154 
155  end subroutine make_gravity
