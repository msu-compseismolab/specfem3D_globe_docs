28 ! add_model_globe_iso
29 !
30 ! this program can be used to update ISOTROPIC model files with
31 ! (smoothed & summed) event kernels.
32 ! the kernels are given for isotropic parameters (alpha,beta,rho) or ( bulk_c,beta,rho).
33 !
34 ! the algorithm uses a steepest descent method with a step length
35 ! determined by the given maximum update percentage.
36 !
37 ! input:
38 ! - step_fac : step length to update the models, f.e. 0.03 for plusminus 3%
39 !
40 ! setup:
41 !
42 !- INPUT_MODEL/ contains:
43 ! proc000***_reg1_vs.bin &
44 ! proc000***_reg1_vp.bin &
45 ! proc000***_reg1_rho.bin
46 !
47 !- INPUT_GRADIENT/ contains:
48 ! proc000***_reg1_bulk_c_kernel_smooth.bin &
49 ! proc000***_reg1_bulk_beta_kernel_smooth.bin &
50 ! proc000***_reg1_rho_kernel_smooth.bin
51 ! or
52 ! proc000***_reg1_alpha_kernel_smooth.bin &
53 ! proc000***_reg1_beta_kernel_smooth.bin &
54 ! proc000***_reg1_rho_kernel_smooth.bin
55 !
56 !- topo/ contains:
57 ! proc000***_reg1_solver_data.bin
58 !
59 ! new models are stored in
60 !- OUTPUT_MODEL/ as
61 ! proc000***_reg1_vp_new.bin and
62 ! proc000***_reg1_vs_new.bin and
63 ! proc000***_reg1_rho_new.bin and
64 !
65 ! USAGE: e.g. ./add_model_globe_iso 0.3
66

68 program add_model
69 
70  use tomography_model_iso
71  use tomography_kernels_iso
72 
73  implicit none
74  integer :: i,j,k,ispec,ier
75  real(kind=CUSTOM_REAL) :: beta1,beta0,rho1,rho0,alpha1,alpha0
76  real(kind=CUSTOM_REAL) :: dbetaiso,dbulk
77 
78  ! ============ program starts here =====================
79 
80  ! initializes arrays
81  call initialize()
82 
83  ! reads in parameters needed
84  call read_parameters_tomo()
85 
86  ! user output
87  if (myrank == 0) then
88  print *
89  print *,'***********'
90  print *,'program add_model_iso: '
91  print *,' NPROC_XI , NPROC_ETA: ',nproc_xi_val,nproc_eta_val
92  print *,' NCHUNKS: ',nchunks_val
93  print *
94  print *,'model update for vs & vp & rho'
95  print *,' step_fac = ',step_fac
96  print *
97  if (use_alpha_beta_rho) then
98  print *,'kernel parameterization: (alpha,beta,rho)'
99  else
100  print *,'kernel parameterization: (bulk,beta,rho)'
101  endif
102  print *
103  if (use_rho_scaling) then
104  print *,'scaling rho perturbations'
105  print *
106  endif
107  print *,'***********'
108  print *
109  endif
110 
111  ! reads in current isotropic model files: vp & vs & rho
112  call read_model_iso()
113 
114  ! reads in smoothed kernels: bulk, beta, rho
115  call read_kernels_iso()
116 
117  ! calculates gradient
118  ! steepest descent method
119  call get_gradient_steepest_iso()
120 
121  ! computes new model values for alpha, beta and rho
122  ! and stores new model files
123  ! allocate new model arrays
124  allocate(model_vp_new(ngllx,nglly,ngllz,nspec), &
125  model_vs_new(ngllx,nglly,ngllz,nspec), &
126  model_rho_new(ngllx,nglly,ngllz,nspec),stat=ier)
127  if (ier /= 0) stop 'Error allocating model arrays'
128 
129  ! initializes arrays
130  model_vp_new = 0.0_custom_real
131  model_vs_new = 0.0_custom_real
132  model_rho_new = 0.0_custom_real
133 
134  ! model update:
135  ! isotropic update everywhere
136  do ispec = 1, nspec
137  do k = 1, ngllz
138  do j = 1, nglly
139  do i = 1, ngllx
140 
141  ! initial model values
142  beta0 = model_vs(i,j,k,ispec)
143  rho0 = model_rho(i,j,k,ispec)
144  alpha0 = model_vp(i,j,k,ispec)
145 
146  beta1 = 0._custom_real
147  rho1 = 0._custom_real
148  alpha1 = 0._custom_real
149 
150  ! isotropic model update
151 
152  ! shear values
153  dbetaiso = model_dbeta(i,j,k,ispec)
154  beta1 = beta0 * exp( dbetaiso )
155 
156  ! density
157  rho1 = rho0 * exp( model_drho(i,j,k,ispec) )
158 
159  ! alpha values
160  dbulk = model_dbulk(i,j,k,ispec)
161  if (use_alpha_beta_rho) then
162  ! new vp values use alpha model update
163  alpha1 = alpha0 * exp( dbulk )
164  else
165  ! new vp values use bulk model update:
166  ! this is based on vp_new = sqrt( bulk_new**2 + 4/3 vs_new**2 )
167  alpha1 = sqrt( alpha0**2 * exp(2.0*dbulk) + four_thirds * beta0**2 * ( &
168  exp(2.0*dbetaiso) - exp(2.0*dbulk) ) )
169  endif
170 
171  ! stores new model values
172  model_vp_new(i,j,k,ispec) = alpha1
173  model_vs_new(i,j,k,ispec) = beta1
174  model_rho_new(i,j,k,ispec) = rho1
175 
176  enddo
177  enddo
178  enddo
179  enddo
180  call synchronize_all()
181 
182  ! stores new model in files
183  call write_new_model_iso()
184 
185  ! stores relative model perturbations
186  call write_new_model_perturbations_iso()
187 
188  ! computes volume element associated with points, calculates kernel integral for statistics
189  call compute_kernel_integral_iso()
190 
191  ! stop all the MPI processes, and exit
192  call finalize_mpi()
193 
194 end program add_model
