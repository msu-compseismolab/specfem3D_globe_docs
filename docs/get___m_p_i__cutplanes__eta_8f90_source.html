34 ! this routine detects cut planes along eta
35 ! In principle the left cut plane of the first slice
36 ! and the right cut plane of the last slice are not used
37 ! in the solver except if we want to have periodic conditions
38 
39  use constants
40 
41  implicit none
42 
43  integer :: nspec
44 
45  logical,dimension(2,nspec) :: iMPIcut_eta
46 
47  integer,dimension(NGLLX,NGLLY,NGLLZ,nspec) :: ibool
48 
49  double precision,dimension(NGLLX,NGLLY,NGLLZ,nspec) :: xstore,ystore,zstore
50 
51  ! logical mask used to create arrays iboolleft_eta and iboolright_eta
52  integer :: npointot
53  logical,dimension(npointot) :: mask_ibool
54 
55  integer, dimension(MAX_NUM_REGIONS,NB_SQUARE_EDGES_ONEDIR) :: NSPEC2D_XI_FACE
56 
57  integer :: iregion

64

67

75 
76  ! debug: file output

78 
79  ! theoretical number of surface elements in the buffers
80  ! cut planes along eta=constant correspond to XI faces
81  nspec2dtheor = nspec2d_xi_face(iregion,1)
82 
83 ! write the MPI buffers for the left and right edges of the slice
84 ! and the position of the points to check that the buffers are fine
85 
86 !
87 ! determine if the element falls on the left MPI cut plane
88 !
89 
90  if (debug) then
91  ! global point number and coordinates left MPI cut-plane
92  open(unit=iout,file=prname(1:len_trim(prname))//'iboolleft_eta.txt',status='unknown')
93  endif
94 
95  ! erase the logical mask used to mark points already found
96  mask_ibool(:) = .false.
97 
98  ! nb of global points shared with the other slice
99  iboolleft_eta(:) = 0
100  npoin2d_eta = 0
101  npoin2d_eta_all(1) = 1
102 
103  ! nb of elements in this cut-plane
104  ispecc1 = 0
105 
106  do ispec = 1,nspec
107  if (impicut_eta(1,ispec)) then
108  ispecc1=ispecc1+1
109  ! loop on all the points in that 2-D element, including edges
110  iy = 1
111  do ix = 1,ngllx
112  do iz = 1,ngllz
113  ! select point, if not already selected
114  if (.not. mask_ibool(ibool(ix,iy,iz,ispec))) then
115  mask_ibool(ibool(ix,iy,iz,ispec)) = .true.
116  npoin2d_eta = npoin2d_eta + 1
117 
118  ! fills buffer arrays
119  iboolleft_eta(npoin2d_eta) = ibool(ix,iy,iz,ispec)
120 
121  npoin2d_eta_all(1) = npoin2d_eta_all(1) + 1
122 
123  ! debug file output
124  if (debug) then
125  write(iout,*) ibool(ix,iy,iz,ispec), xstore(ix,iy,iz,ispec), &
126  ystore(ix,iy,iz,ispec),zstore(ix,iy,iz,ispec)
127  endif
128  endif
129  enddo
130  enddo
131  endif
132  enddo
133 
134  if (debug) then
135  ! put flag to indicate end of the list of points
136  write(iout,*) '0 0 0. 0. 0.'
137  ! write total number of points
138  write(iout,*) npoin2d_eta
139  close(iout)
140  endif
141 
142 ! compare number of surface elements detected to analytical value
143  if (ispecc1 /= nspec2dtheor) call exit_mpi(myrank,'Error MPI cut-planes detection in eta=left')
144 
145  ! subtract the line that contains the flag after the last point
146  npoin2d_eta_all(1) = npoin2d_eta_all(1) - 1
147  if (npoin2d_eta_all(1) > nglob2dmax_ymin_ymax .or. npoin2d_eta_all(1) /= npoin2d_eta) &
148  call exit_mpi(myrank,'incorrect iboolleft_eta read')
149 
150

155 
156  if (debug) then
157  ! global point number and coordinates right MPI cut-plane
158  open(unit=iout,file=prname(1:len_trim(prname))//'iboolright_eta.txt',status='unknown')
159  endif
160 
161  ! erase the logical mask used to mark points already found
162  mask_ibool(:) = .false.
163 
164  ! nb of global points shared with the other slice
165  iboolright_eta(:) = 0

167  npoin2d_eta_all(2) = 1
168 
169  ! nb of elements in this cut-plane
170  ispecc2 = 0
171 
172  do ispec = 1,nspec
173  if (impicut_eta(2,ispec)) then
174  ispecc2=ispecc2+1
175  ! loop on all the points in that 2-D element, including edges
176  iy = nglly
177  do ix = 1,ngllx
178  do iz = 1,ngllz
179  ! select point, if not already selected
180  if (.not. mask_ibool(ibool(ix,iy,iz,ispec))) then
181  mask_ibool(ibool(ix,iy,iz,ispec)) = .true.
182  npoin2d_eta = npoin2d_eta + 1

186 
187  npoin2d_eta_all(2) = npoin2d_eta_all(2) + 1
188 
189  ! debug file output
190  if (debug) then
191  write(iout,*) ibool(ix,iy,iz,ispec), xstore(ix,iy,iz,ispec), &
192  ystore(ix,iy,iz,ispec),zstore(ix,iy,iz,ispec)
193  endif
194  endif
195  enddo
196  enddo
197  endif
198  enddo
199 
200  if (debug) then
201  ! put flag to indicate end of the list of points
202  write(iout,*) '0 0 0. 0. 0.'
203  ! write total number of points
204  write(iout,*) npoin2d_eta
205  close(iout)
206  endif
207 
208  ! compare number of surface elements detected to analytical value
209  if (ispecc2 /= nspec2dtheor) call exit_mpi(myrank,'Error MPI cut-planes detection in eta=right')
210 
211  ! subtract the line that contains the flag after the last point
212  npoin2d_eta_all(2) = npoin2d_eta_all(2) - 1
213  if (npoin2d_eta_all(2) > nglob2dmax_ymin_ymax .or. npoin2d_eta_all(2) /= npoin2d_eta) &
214  call exit_mpi(myrank,'incorrect iboolright_eta read')
215 
216  end subroutine get_mpi_cutplanes_eta
