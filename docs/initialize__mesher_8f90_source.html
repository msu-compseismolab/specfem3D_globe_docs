28  subroutine initialize_mesher()
29 
30  use meshfem3d_par
31  use meshfem3d_models_par
32 
33  use manager_adios

35  implicit none

39  ! local parameters
40  integer :: sizeprocs
41  ! timing
42  double precision, external :: wtime
43 
44 ! sizeprocs returns number of processes started (should be equal to NPROCTOT).
45 ! myrank is the rank of each process, between 0 and NPROCTOT-1.
46 ! as usual in MPI, process 0 is in charge of coordinating everything
47 ! and also takes care of the main output
48 ! do not create anything for the inner core here, will be done in solver
49  call world_size(sizeprocs)
50  call world_rank(myrank)
51 
52 ! set the base pathname for output files
53  output_files = output_files_base
54 
55 ! open main output file, only written to by process 0
56  if (myrank == 0) then
57  if (imain /= istandard_output) &
58  open(unit=imain,file=trim(output_files)//'/output_mesher.txt',status='unknown')
59 
60  write(imain,*)
61  write(imain,*) '****************************'
62  write(imain,*) '*** Specfem3D MPI Mesher ***'
63  write(imain,*) '****************************'
64  write(imain,*)
65  write(imain,*) 'Version: ', git_version
66  write(imain,*)
67  call flush_imain()
68  endif
69 
70 ! get MPI starting time
71  time_start = wtime()
72 
73  if (myrank == 0) then
74  ! reads the parameter file and computes additional parameters
75  call read_compute_parameters()
76  endif
77 
78  ! broadcast parameters read from master to all processes
79  call broadcast_computed_parameters()
80 
81  ! check that the code is running with the requested number of processes
82  if (sizeprocs /= nproctot) then
83  if (myrank == 0) print *,'Error wrong number of MPI processes ',sizeprocs,' should be ',nproctot,', please check...'
84  call exit_mpi(myrank,'wrong number of MPI processes')
85  endif
86 
87  ! synchronizes processes
88  call synchronize_all()
89 
90  ! compute rotation matrix from Euler angles
91  angular_width_xi_rad = angular_width_xi_in_degrees * degrees_to_radians
92  angular_width_eta_rad = angular_width_eta_in_degrees * degrees_to_radians
93 
94  if (nchunks /= 6) call euler_angles(rotation_matrix,center_longitude_in_degrees,center_latitude_in_degrees,gamma_rotation_azimuth)
95 
96  ! ADIOS
97  if (adios_enabled) then
98  call initialize_adios()
99  endif
100 
101  ! gravity integrals
102  if (gravity_integrals) then
103  call gravity_initialize_integrals()
104  endif
105 
106  ! OpenMP
107  call init_openmp()
108 
109  end subroutine initialize_mesher
