30 ! convert x y z to r theta phi, single precision call
31 
32  use constants, only: custom_real,size_real,small_val_angle,zero
33 
34  implicit none
35 
36  real(kind=CUSTOM_REAL), intent(in) :: x,y,z
37  real(kind=CUSTOM_REAL), intent(out) :: r,theta,phi
38 
39  double precision xmesh,ymesh,zmesh
40 
41  xmesh = dble(x)
42  ymesh = dble(y)
43  zmesh = dble(z)
44 
45  if (zmesh > -small_val_angle .and. zmesh <= zero) zmesh = -small_val_angle
46  if (zmesh < small_val_angle .and. zmesh >= zero) zmesh = small_val_angle
47

49 
50  if (xmesh > -small_val_angle .and. xmesh <= zero) xmesh = -small_val_angle
51  if (xmesh < small_val_angle .and. xmesh >= zero) xmesh = small_val_angle
52 
53  phi = real(datan2(ymesh,xmesh), kind=custom_real)
54 
55  r = real(dsqrt(xmesh*xmesh + ymesh*ymesh + zmesh*zmesh), kind=custom_real)
56 
57  end subroutine xyz_2_rthetaphi

113  subroutine xyz_2_rlatlon_dble(x,y,z,r,lat,lon)
114 
115 ! converts geocentric coordinates x/y/z to geographic radius/latitude/longitude (in degrees)
116 
117  use constants, only: radians_to_degrees,pi_over_two
118 
119  implicit none
120 
121  double precision,intent(in) :: x,y,z
122  double precision,intent(out) :: r,lat,lon
123 
124  ! local parameters
125  double precision :: theta,phi,theta_prime
126 
127  ! converts location to radius/colatitude/longitude
128  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
129 
130  ! reduces range for colatitude to 0 and PI, for longitude to 0 and 2*PI
131  call reduce(theta,phi)
132 
133  ! converts geocentric to geographic colatitude
134  ! note: for example, the moho/topography/3D-model information is given in geographic latitude/longitude
135  ! (lat/lon given with respect to a reference ellipsoid).
136  ! we need to convert the geocentric mesh positions (theta,phi) to geographic ones (lat/lon),
137  ! thus correcting geocentric latitude for ellipticity
138  call geocentric_2_geographic_dble(theta,theta_prime)
139 
140  ! gets geographic latitude and longitude in degrees
141  lat = (pi_over_two - theta_prime) * radians_to_degrees
142  lon = phi * radians_to_degrees
143 
144  end subroutine xyz_2_rlatlon_dble

148  subroutine geocentric_2_geographic_dble(theta,theta_prime)
149 
150 ! converts geocentric colatitude (theta) to geographic colatitude (theta_prime) (in radians)
151 
152 ! see the book of Dahlen and Tromp, 1998: chapter 14.1.5 Geographic versus geocentric colatitude
153 
154 ! (see also in file geographic_versus_geocentric_coordinates_from_Dahlen_Tromp_1998.pdf in the "doc" directory of the code)
155 
156 !
157 ! The location of a seismic station situated upon the Earth's surface is conventionally specified
158 ! by giving its elevation (e) above the geoid, its longitude (phi) with respect to Greenwich,
159 ! and its geographical colatitude (theta_prime), which is the angle between the normal n to the reference ellipsoid
160 ! and the normal z to the equatorial plane. (...)
161 ! It is necessary to convert the geographic colatitude (theta_prime) to the corresponding geocentric colatitude (theta),
162 ! which is the angle between the radius vector r and z.
163 !
164 !
165 ! @jeroen:
166 ! "
167 ! The conversion is given in the book of Dahlen and Tromp, 1998 (equation 14.32),
168 ! and using the observed ellipticity given by (14.24) this gives 1.00670466 (not 1.006760466).
169 ! If you use the hydrostatic value given by (14.23),
170 ! you would get 1.0066711.
171 ! One could argue for either one, but I prefer to use the best fitting observed value.
172 !
173 ! With regards to the conversion from geographic to geocentric coordinates:
174 ! all station and event locations are reported by bulletins in map coordinates,
175 ! that is, geographic coordinates. Thus the assumption should be that
176 ! locations given in the STATIONS and CMTSOLUTION files are geographic
177 ! coordinates. When you are reading a topographic map, again the
178 ! assumption is that we are dealing with geographic coordinates,
179 ! and those coordinates need to be converted to geocentric coordinates.
180 ! "
181 !
182 ! for WGS84 ellipsoid, eccentricity squared e**2 = 0.00669437999014
183 !
184 ! used here: e**2 = 0.0066600069180170474
185 !
186 ! the geocentric latitude (lat) and geographic latitude (lat_prime) derives as
187 ! tan(lat) = (1 - e**2) * tan(lat_prime) (1)
188 !
189 ! from (1) it is clear that the geographic latitude can be written as
190 ! lat_prime = atan( 1/(1 - e**2) * tan(lat) )
191 !
192 ! since
193 ! latitude lat = PI/2 - theta
194 ! with theta being colatitude and
195 ! tan( PI/2 - theta ) = cot( theta ) = 1/tan(theta) = cos(theta)/sin(theta)
196 !
197 ! we find that the geographic colatitude (theta_prime) and geocentric colatitude (theta) relate as
198 ! tan(theta_prime) = ( 1 - e**2 ) tan(theta)
199 !
200 ! or using the latitude expressions from above, we have the geographic colatitude (theta_prime) as
201 ! theta_prime = PI/2 - atan( 1/(1 - e**2) * 1/tan(theta) )
202 ! = PI/2 - atan( 1/(1 - e**2) * cos(theta)/sin(theta) )
203 ! = PI/2 - atan( 1.00670466 * cos(theta)/sin(theta) )
204

210 ! 14.1.5 Geographic versus geocentric colatitude
211

213 ! by giving its elevation (e) above the geoid, its longitude (phi) with respect to Greenwich,
214 ! and its geographical colatitude (theta_prime), which is the angle
215 ! between the normal n to the reference ellipsoid
216 ! and the normal z to the equatorial plane. (...)
217 ! it is necessary to convert the geographic colatitude (theta_prime) to
218 ! the corresponding geocentric colatitude (theta),
219 ! which is the angle between the radius vector r and z.
220 
221 ! Second, a while ago, in our email exchange with Jeroen about this, he states:
222 
223 ! "
224 ! The conversion is given by D & T (14.32), and using the observed ellipticity given by (14.24)
225 ! this gives 1.00670466 (not 1.006760466).
226 ! If you use the hydrostatic value given by (14.23),
227 ! you would get 1.0066711.
228 ! One could argue for either one, but I prefer to use the best fitting observed value.
229 !
230 ! With regards to the conversion from geographic to geocentric coordinates:
231 ! all station and event locations are reported by bulletins in map coordinates,
232 ! that is, geographic coordinates. Thus the assumption should be that
233 ! locations given in the STATIONS and CMTSOLUTION files are geographic
234 ! coordinates. When you are reading a topographic map, again the
235 ! assumption is that we are dealing with geographic coordinates,
236 ! and those coordinates need to be converted to geocentric coordinates.
237 ! "
238 
239 ! It is a tricky thing to have a correct way of doing this, but I think we
240 ! are coherent when dealing with it, also in case ellipticity is turned
241 ! on. here some points to consider:
242 
243 ! mesher:
244 
245 ! - topography:
246 ! topography is always added on top of a spherical mesh
247 ! -> the geocentric mesh positions are converted to geographic ones to
248 ! find the corresponding elevation
249 
250 ! - ellipticity:
251 ! is added at the very end of the compute_element_** routines, i.e. after
252 ! assigning mantle properties and topography.
253 ! it stretches out all mesh nodes in radial direction.
254 ! -> the ellipticity factor is based on the geocentric colatitude of the
255 ! initial mesh node location.
256 
257 ! I think in the mesher the conversions are all fine and we don't need to
258 ! check for ELLIPTICITY in the conversion routines. topography data is in
259 ! general given in geographic positions, so we need to convert from
260 ! geocentric to geographic locations, no matter what ellipticity flag we
261 ! have set. in that sense, the flag ASSUME_PERFECT_SPHERE is overriding
262 ! this assumption, i.e. then we assume that topography data would be given
263 ! at geocentric positions.
264 
265 ! The tricky part comes in the solver:
266 
267 ! solver:
268 
269 ! - moment tensor:
270 ! the moment tensor of the CMTSOLUTION is rotated from a spherical to a
271 ! Cartesian representation, using the geocentric colatitude
272 ! - vertical component:
273 ! the vertical component at source/receiver locations is calculated with
274 ! respect to geocentric colatitude
275 ! - target position in the mesh:
276 ! to find the best source/receiver position in the mesh, the target
277 ! location is using the geocentric colatitude and corrects the radius for
278 ! topography and ellipticity when calculating the x/y/z Cartesian
279 ! position. We then locate this x/y/z location in the mesh.
280 
281 ! Note that we always convert the geographic latitude to geocentric
282 ! colatitude for source/station positions, unless ASSUME_PERFECT_SPHERE is
283 ! set. if that flag is set, we assume that source/station locations are
284 ! given in geocentric positions already.
285 
286 ! I think the confusing part is for the elliptical mesh. in case of a
287 ! spherical mesh, we would locate the geocentric position in the spherical
288 ! mesh, which should be fine. however, for the elliptical case where
289 ! ELLIPTICITY is turn on, the mesh would be stretched out and x/y/z
290 ! positions of the mesh nodes are then what? geocentric or geographic now?
291 ! and what is the Cartesian position of a location given in geographic
292 ! coordinates? since we map topography onto geocentric colatitude and
293 ! stretch out the ellipticity in radial direction, i think we always have
294 ! to convert first the source/receiver positions to geocentric ones to
295 ! then locate the best Cartesian x/y/z location in the mesh. so, i think,
296 ! we are doing it the right way.
297 
298 ! Not sure if i made this point clear, and i would welcome some more
299 ! thoughts about the rotations. the rotations for components and moment
300 ! tensors are using a vertical based on the geocentric colatitude? i'm not
301 ! sure if this is correct. so, my question would rather be: what is
302 ! vertical for a seismometer?
303 
304 ! I would think that when you install one, you would use the plumb line to
305 ! orient it. this would mean that you are orienting it along the local
306 ! vertical, which is always perpendicular to the geoid. i guess you know
307 ! that the geoid looks like a potato and not like a perfect sphere, so
308 ! vertical on a seismometer seems not to be the same like the radial
309 ! direction in a geocentric framework...
310 
311 ! Anyway, I'm sorry that I'm not giving a clear answer, but hopefully this
312 ! gives some thoughts.

320 
321  ! note: september, 2014
322  ! factor: 1/(1 - e^2) = 1/(1 - (1 - (1-f)^2)) = 1/( (1-f)^2 )
323  ! with eccentricity e^2 = 1 - (1-f)^2
324  ! see about Earth flattening in constants.h: flattening factor changed to 1/299.8
325  ! f = 1/299.8 -> 1/( (1-f)^2 ) = 1.0067046409645724
326  double precision, parameter :: FACTOR_TAN = 1.d0 / one_minus_f_squared

329 
330  if (.not. assume_perfect_sphere) then
331  ! mesh is elliptical
332  if (use_old_version_5_1_5_format) then
333  theta_prime = pi_over_two - datan(1.006760466d0*dcos(theta)/dmax1(tinyval,dsin(theta)))
334  else

336  theta_prime = pi_over_two - datan(factor_tan*dcos(theta)/dmax1(tinyval,dsin(theta)))

376  subroutine lat_2_geocentric_colat_dble(lat_prime,theta)
377 
378 ! converts geographic latitude (lat_prime) (in degrees) to geocentric colatitude (theta) (in radians)
379 
380  use constants, only: pi_over_two,degrees_to_radians,assume_perfect_sphere,one_minus_f_squared
381 
382  implicit none
383 
384  ! latitude (in degrees)
385  double precision,intent(in) :: lat_prime
386  ! co-latitude (in radians)
387  double precision,intent(out) :: theta
388 
389  if (.not. assume_perfect_sphere) then
390  ! converts geographic (lat_prime) to geocentric latitude and converts to co-latitude (theta)
391  theta = pi_over_two - atan( one_minus_f_squared*dtan(lat_prime * degrees_to_radians) )

394  ! converts latitude (in degrees to co-latitude (in radians)
395  theta = pi_over_two - lat_prime * degrees_to_radians
396  endif

398  end subroutine lat_2_geocentric_colat_dble
