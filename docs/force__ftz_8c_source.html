41 /* Flush-To-Zero (FTZ) mode */
42 
43 /* FTZ is a method of bypassing IEEE 754 methods of dealing with */
44 /* invalid floating-point numbers due to underflows. This mode is much */
45 /* faster. Two conditions must be met for FTZ processing to occur: */
46 
47 /* * The FTZ bit (bit 15) in the MXCSR register must be masked (value = 1). */
48 /* * The underflow exception (bit 11) needs to be masked (value = 1). */
49 
50 /* This routine is not strictly necessary for SPECFEM, thus if it does not compile on your system
51  (since it calls some low-level system routines) just suppress all the lines below (i.e. make it an empty file)
52  and comment out the call to force_ftz() in the main SPECFEM program */
53 
54 #include "config.h"
55 
56 #define FTZ_BIT 15
57 #define UNDERFLOW_EXCEPTION_MASK 11
58 
59 #ifdef HAVE_XMMINTRIN
60  #define FORCE_FTZ
61  #include <xmmintrin.h>
62 #elif HAVE_EMMINTRIN
63  #include <emmintrin.h>
64  #define FORCE_FTZ
65 #endif

67 void
68 FC_FUNC_(force_ftz,FORCE_FTZ)()
69 {
70 #ifdef FORCE_FTZ
71  unsigned int x;
72 
73  /* force FTZ by setting bits 11 and 15 to one */
74  x = _mm_getcsr();
75  x |= (1 << FTZ_BIT);
76  x |= (1 << UNDERFLOW_EXCEPTION_MASK);
77  _mm_setcsr(x);
78 #endif
79 }
