73 !
74 !---- read info
75 !
76  forcesolution = 'DATA/FORCESOLUTION'
77 ! see if we are running several independent runs in parallel
78 ! if so, add the right directory for that run
79 ! (group numbers start at zero, but directory names start at run0001, thus we add one)
80 ! a negative value for "mygroup" is a convention that indicates that groups (i.e. sub-communicators, one per run) are off
81  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
82  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
83  forcesolution = path_to_add(1:len_trim(path_to_add))//forcesolution(1:len_trim(forcesolution))
84  endif
85 
86  open(unit=iin,file=trim(forcesolution),status='old',action='read',iostat=ier)
87  if (ier /= 0) then
88  print *,'Error opening file: ',trim(forcesolution)
89  stop 'Error opening FORCESOLUTION file'
90  endif
91 
92 ! read source number isource
93  do isource=1,nsources
94 
95  read(iin,"(a)") string
96  ! skips empty lines
97  do while (len_trim(string) == 0)
98  read(iin,"(a)") string
99  enddo
100 
101  ! read header with event information
102  read(string,"(a6,i4)") dummy,dummyval
103 
104  ! read time shift
105  read(iin,"(a)") string
106  read(string(12:len_trim(string)),*) t_shift(isource)
107 
108  ! read f0 (stored in hdur() array for convenience, to use the same array as for CMTSOLUTION)
109  ! Please be careful, if you meet an error in reading the file FORCESOLUTION,
110  ! such as you still write "hdur:" instead of "f0:"
111  ! Please change your file or do following change in the code, such as changing
112  ! read(string(4:len_trim(string)),*) hdur(isource)
113  ! to
114  ! read(string(6:len_trim(string)),*) hdur(isource)
115  read(iin,"(a)") string

142  ! read direction vector's North component
143  read(iin,"(a)") string
144  read(string(29:len_trim(string)),*) comp_dir_vect_source_n(isource)
145 
146  ! read direction vector's vertical component
147  read(iin,"(a)") string
148  read(string(32:len_trim(string)),*) comp_dir_vect_source_z_up(isource)
149 
150  ! checks half-duration
151  if (force_stf(isource) == 0) then
152  ! Step source time function
153  ! null half-duration indicates a Heaviside
154  ! replace with very short error function
155  if (hdur(isource) < 5. * dt ) hdur(isource) = 5. * dt
156  else if (force_stf(isource) == 1) then
157  ! Ricker source time function
158  ! half-duration is the dominant frequency for the
159  ! null half-duration indicates a very low-frequency source
160  ! (see constants.h: TINYVAL = 1.d-9 )
161  if (hdur(isource) < tinyval ) hdur(isource) = tinyval
162  else
163  stop 'unsupported force_stf value!'
164  endif
165 
166  enddo
167 
168  close(iin)
169 
170  ! Sets tshift_force to zero to initiate the simulation!
171  if (nsources == 1) then
172  tshift_force = 0.d0
173  min_tshift_force_original = t_shift(1)
174  else
175  tshift_force(1:nsources) = t_shift(1:nsources)-minval(t_shift)
176  min_tshift_force_original = minval(t_shift)
177  endif
178 
179  do isource=1,nsources
180  ! checks half-duration
181  ! half-duration is the dominant frequency of the source
182  ! point forces use a Ricker source time function
183  ! null half-duration indicates a very low-frequency source
184  ! (see constants.h: TINYVAL = 1.d-9 )
185  if (hdur(isource) < tinyval) hdur(isource) = tinyval
186 
187  ! check (tilted) force source direction vector
188  length = sqrt( comp_dir_vect_source_e(isource)**2 + comp_dir_vect_source_n(isource)**2 + &
189  comp_dir_vect_source_z_up(isource)**2)
190  if (length < tinyval) then
191  print *, 'normal length: ', length
192  print *, 'isource: ',isource
193  stop 'error set force point normal length, make sure all forces have a non null direction vector'
194  endif
195  enddo
196 
197  ! scale and non-dimensionalize the factor_force_source
198  ! factor_force_source in FORCESOLUTION file is in Newton
199  ! 1 Newton is 1 kg * 1 m / (1 second)^2
200  !
201  scalef = rhoav * (r_earth**4) * pi*grav*rhoav
202  factor_force_source(:) = factor_force_source(:) / scalef
