28 ! add_model_globe_tiso_cg
29 !
30 ! this program can be used to update TRANSVERSE ISOTROPIC model files
31 ! based on smoothed event kernels.
32 ! the kernels are given for tranverse isotropic parameters (bulk_c,bulk_betav,bulk_betah,eta).
33 !
34 ! the algorithm uses a conjugate gradient method with a step length
35 ! limited by the given maximum update percentage.
36 !
37 ! input:
38 ! - step_fac : step length to update the models, f.e. 0.03 for plusminus 3%
39 !
40 ! setup:
41 !
42 !- INPUT_MODEL/ contains:
43 ! proc000***_reg1_vsv.bin &
44 ! proc000***_reg1_vsh.bin &
45 ! proc000***_reg1_vpv.bin &
46 ! proc000***_reg1_vph.bin &
47 ! proc000***_reg1_eta.bin &
48 ! proc000***_reg1_rho.bin
49 !
50 !- INPUT_GRADIENT/ contains:
51 ! proc000***_reg1_bulk_c_kernel_smooth.bin &
52 ! proc000***_reg1_bulk_betav_kernel_smooth.bin &
53 ! proc000***_reg1_bulk_betah_kernel_smooth.bin &
54 ! proc000***_reg1_eta_kernel_smooth.bin &
55 ! proc000***_reg1_rho_kernel_smooth.bin
56 !
57 !- /tigress-hsm/dpeter/SPECFEM3D_GLOBE/KERNELS/OUTPUT_SUM.old/ contains old gradients:
58 ! proc000***_reg1_bulk_c_kernel_smooth.bin &
59 ! proc000***_reg1_bulk_betav_kernel_smooth.bin &
60 ! proc000***_reg1_bulk_betah_kernel_smooth.bin &
61 ! proc000***_reg1_eta_kernel_smooth.bin &
62 ! proc000***_reg1_rho_kernel_smooth.bin
63 !
64 !- topo/ contains:
65 ! proc000***_reg1_solver_data_1.bin
66 !
67 ! new models are stored in
68 !- OUTPUT_MODEL/ as
69 ! proc000***_reg1_vpv_new.bin &
70 ! proc000***_reg1_vph_new.bin &
71 ! proc000***_reg1_vsv_new.bin &
72 ! proc000***_reg1_vsh_new.bin &
73 ! proc000***_reg1_eta_new.bin &
74 ! proc000***_reg1_rho_new.bin
75 !
76 ! usage: ./add_model_globe_tiso_cg 0.3
77 !
78 !
79 ! NOTE: this routine uses previous model update in OUTPUT_SUM.old/
80 ! for a conjugate gradient update
81 !

115  print *,'***********'
116  print *

129  ! conjugate gradient method

133  ! (see also Carl's Latex notes)
134  ! allocate new model arrays

136  model_vph_new(ngllx,nglly,ngllz,nspec), &
137  model_vsv_new(ngllx,nglly,ngllz,nspec), &
138  model_vsh_new(ngllx,nglly,ngllz,nspec), &
139  model_eta_new(ngllx,nglly,ngllz,nspec), &
140  model_rho_new(ngllx,nglly,ngllz,nspec),stat=ier)
141  if (ier /= 0) stop 'Error allocating model arrays'

148  model_eta_new = 0.0_custom_real
149  model_rho_new = 0.0_custom_real

152  ! transverse isotropic update only in layer Moho to 220 (where SPECFEM3D_GLOBE considers TISO)
153  ! everywhere else uses an isotropic update
154  do ispec = 1, nspec
155  do k = 1, ngllz
156  do j = 1, nglly
157  do i = 1, ngllx
158

160  eta0 = model_eta(i,j,k,ispec)
161  betav0 = model_vsv(i,j,k,ispec)
162  betah0 = model_vsh(i,j,k,ispec)
163  rho0 = model_rho(i,j,k,ispec)
164  alphav0 = model_vpv(i,j,k,ispec)
165  alphah0 = model_vph(i,j,k,ispec)
166 
167  eta1 = 0._custom_real
168  betav1 = 0._custom_real
169  betah1 = 0._custom_real
170  rho1 = 0._custom_real
171  alphav1 = 0._custom_real
172  alphah1 = 0._custom_real
173 
174  ! do not use transverse isotropy except if element is between d220 and Moho
175 ! if (.not. ( idoubling(ispec)==IFLAG_220_80 .or. idoubling(ispec)==IFLAG_80_MOHO)) then
176  if (.not. ispec_is_tiso(ispec)) then
177  ! isotropic model update
178 
179  ! no eta perturbation, since eta = 1 in isotropic media
180  eta1 = eta0
181 
182  ! shear values
183  ! isotropic kernel K_beta = K_betav + K_betah
184  ! with same scaling step_length the model update dbeta_iso = dbetav + dbetah
185  ! note:
186  ! this step length can be twice as big as that given by the input
187  dbetaiso = model_dbetav(i,j,k,ispec) + model_dbetah(i,j,k,ispec)
188  betav1 = betav0 * exp( dbetaiso )
189  betah1 = betah0 * exp( dbetaiso )
190  ! note: betah is probably not really used in isotropic layers
191  ! (see SPECFEM3D_GLOBE/get_model.f90)
192 
193  ! density: uses scaling relation with isotropic shear perturbations
194  ! dln rho = RHO_SCALING * dln betaiso
195  rho1 = rho0 * exp( rho_scaling * dbetaiso )
196 
197  ! alpha values
198  dbulk = model_dbulk(i,j,k,ispec)
199  alphav1 = sqrt( alphav0**2 * exp(2.0*dbulk) + four_thirds * betav0**2 * ( &
200  exp(2.0*dbetaiso) - exp(2.0*dbulk) ) )
201  alphah1 = sqrt( alphah0**2 * exp(2.0*dbulk) + four_thirds * betah0**2 * ( &
202  exp(2.0*dbetaiso) - exp(2.0*dbulk) ) )
203  ! note: alphah probably not used in SPECFEM3D_GLOBE
204 
205  else
206 
207  ! transverse isotropic model update
208 
209  ! eta value : limits updated values for eta range constraint
210  eta1 = eta0 * exp( model_deta(i,j,k,ispec) )
211  if (eta1 < limit_eta_min ) eta1 = limit_eta_min
212  if (eta1 > limit_eta_max ) eta1 = limit_eta_max
213 
214  ! shear values
215  betav1 = betav0 * exp( model_dbetav(i,j,k,ispec) )
216  betah1 = betah0 * exp( model_dbetah(i,j,k,ispec) )
217 
218  ! density: uses scaling relation with Voigt average of shear perturbations
219  betaiso0 = sqrt( ( 2.0 * betav0**2 + betah0**2 ) / 3.0 )
220  betaiso1 = sqrt( ( 2.0 * betav1**2 + betah1**2 ) / 3.0 )
221  dbetaiso = log( betaiso1 / betaiso0 )
222  rho1 = rho0 * exp( rho_scaling * dbetaiso )
223 
224  ! alpha values
225  dbulk = model_dbulk(i,j,k,ispec)
226  alphav1 = sqrt( alphav0**2 * exp(2.0*dbulk) &
227  + four_thirds * betav0**2 * ( &
228  exp(2.0*model_dbetav(i,j,k,ispec)) - exp(2.0*dbulk) ) )
229  alphah1 = sqrt( alphah0**2 * exp(2.0*dbulk) &
230  + four_thirds * betah0**2 * ( &
231  exp(2.0*model_dbetah(i,j,k,ispec)) - exp(2.0*dbulk) ) )
232 
233  endif
234 
235 
236  ! stores new model values
237  model_vpv_new(i,j,k,ispec) = alphav1
238  model_vph_new(i,j,k,ispec) = alphah1
239  model_vsv_new(i,j,k,ispec) = betav1
240  model_vsh_new(i,j,k,ispec) = betah1
241  model_eta_new(i,j,k,ispec) = eta1
242  model_rho_new(i,j,k,ispec) = rho1
243 
244  enddo
245  enddo
246  enddo
247  enddo
248 
249  ! stores new model in files
250  call write_new_model_tiso()
251 
252  ! stores relative model perturbations
253  call write_new_model_perturbations_tiso()
254 
255  ! computes volume element associated with points, calculates kernel integral for statistics
256  call compute_kernel_integral_tiso()
257 
258  ! stop all the MPI processes, and exit
259  call finalize_mpi()
260 
261 end program add_model
