29 subroutine read_parameters_tomo()
30 
31 ! reads in parameters needed (only step length for now...)
32 
33  use tomography_par
34 
35  implicit none
36  integer :: ier
37  character(len=MAX_STRING_LEN) :: s_step_fac,arg
38 
39  ! subjective step length to multiply to the gradient
40  ! e.g. step_fac = 0.03
41 
42  call get_command_argument(1,s_step_fac)
43 
44  if (trim(s_step_fac) == '') then
45  call usage()
46  endif
47 
48  ! read in parameter information
49  read(s_step_fac,*,iostat=ier) step_fac
50  if (ier /= 0) then
51  call usage()
52  endif
53 
54  ! safety check
55  if (abs(step_fac) < 1.e-15) then
56  print *,'Error: step factor ',step_fac,' is too small and will lead to no update...'
57  call exit_mpi(myrank,'Error step factor too small')
58  endif
59 
60  ! optional arguments
61  ! input directory which holds (summed) kernel files
62  call get_command_argument(2,arg)
63  if (len_trim(arg) > 0) then
64  if (arg(len_trim(arg):len_trim(arg)) /= '/') then
65  input_kernels_dir = trim(arg) // '/'
66  else
67  input_kernels_dir = trim(arg)
68  endif
69  endif
70 
71  ! output directory for new model files
72  call get_command_argument(3,arg)
73  if (len_trim(arg) > 0) then
74  if (arg(len_trim(arg):len_trim(arg)) /= '/') then
75  output_model_dir = trim(arg) // '/'
76  else
77  output_model_dir = trim(arg)
78  endif
79 
80  ! sets output directory for statistics to same directory
81  if (print_statistics_files) then
82  output_statistics_dir = trim(output_model_dir)
83  endif
84  endif
85 
86  ! statistics
87  if (print_statistics_files .and. myrank == 0) then
88  open(iout,file=trim(output_statistics_dir)//'statistics_step_fac',status='unknown',action='write',iostat=ier)
89  if (ier /= 0) then
90  print *,'Error opening file: ',trim(output_statistics_dir)//'statistics_step_fac'
91  print *,'Please make sure that directory '//trim(output_statistics_dir)//' exists...'
92  print *
93  stop 'Error opening statistics file'
94  endif
95  write(iout,'(1e24.12)') step_fac
96  close(iout)
97  endif
98 
99 contains
100 
101  subroutine usage()
102 
103  implicit none
104 
105  if (myrank == 0) then
106  print *,'Usage: add_model step_factor [INPUT-KERNELS-DIR/] [OUTPUT-MODEL-DIR/]'
107  print *
108  print *,'with'
109  print *,' step_factor - factor to scale gradient (e.g. 0.03 for 3 percent update)'
110  print *,' INPUT-KERNELS-DIR/ - (optional) directory which holds summed kernels (e.g. alpha_kernel.bin,..)'
111  print *,' OUTPUT-MODEL-DIR/ - (optional) directory which will hold new model files (e.g. vp_new.bin,..)'
112  print *
113  print *,'Please rerun e.g. like: mpirun -np ',sizeprocs,' ./bin/xadd_model 0.03'
114  print *
115  endif
116  call synchronize_all()
117  call exit_mpi(myrank,'Error usage: add_model step_factor')
118 
119  end subroutine usage
120 
121 end subroutine read_parameters_tomo
