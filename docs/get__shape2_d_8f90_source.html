28  subroutine get_shape2d(shape2D,dershape2D,xigll,yigll,NGLLA,NGLLB)
29 
30  use constants
31 
32  implicit none
33 
34 ! generic routine that accepts any polynomial degree in each direction
35 
36  integer NGLLA,NGLLB
37 
38  double precision xigll(NGLLA)
39  double precision yigll(NGLLB)
40 
41 ! 2D shape functions and their derivatives
42  double precision shape2D(NGNOD2D,NGLLA,NGLLB)
43  double precision dershape2D(NDIM2D,NGNOD2D,NGLLA,NGLLB)
44 
45  integer i,j,ia
46 
47 ! location of the nodes of the 2D quadrilateral elements
48  double precision xi,eta
49  double precision l1xi,l2xi,l3xi,l1eta,l2eta,l3eta
50  double precision l1pxi,l2pxi,l3pxi,l1peta,l2peta,l3peta
51 
52 ! for checking the 2D shape functions
53  double precision sumshape,sumdershapexi,sumdershapeeta
54 
55 ! check that the parameter file is correct
56  if (ngnod /= 27) call exit_mpi(myrank,'elements should have 27 control nodes')
57  if (ngnod2d /= 9) call exit_mpi(myrank,'surface elements should have 9 control nodes')
58 
59 ! generate the 2D shape functions and their derivatives (9 nodes)
60  do i = 1,nglla
61 
62  xi=xigll(i)
63 
64  l1xi=half*xi*(xi-one)
65  l2xi=one-xi**2
66  l3xi=half*xi*(xi+one)
67 
68  l1pxi=xi-half
69  l2pxi=-two*xi
70  l3pxi=xi+half
71 
72  do j = 1,ngllb
73 
74  eta=yigll(j)
75 
76  l1eta=half*eta*(eta-one)
77  l2eta=one-eta**2
78  l3eta=half*eta*(eta+one)
79 
80  l1peta=eta-half
81  l2peta=-two*eta
82  l3peta=eta+half
83 
84 ! corner nodes
85 
86  shape2d(1,i,j)=l1xi*l1eta
87  shape2d(2,i,j)=l3xi*l1eta
88  shape2d(3,i,j)=l3xi*l3eta
89  shape2d(4,i,j)=l1xi*l3eta
90 
91  dershape2d(1,1,i,j)=l1pxi*l1eta
92  dershape2d(1,2,i,j)=l3pxi*l1eta
93  dershape2d(1,3,i,j)=l3pxi*l3eta
94  dershape2d(1,4,i,j)=l1pxi*l3eta
95 
96  dershape2d(2,1,i,j)=l1xi*l1peta
97  dershape2d(2,2,i,j)=l3xi*l1peta
98  dershape2d(2,3,i,j)=l3xi*l3peta
99  dershape2d(2,4,i,j)=l1xi*l3peta
100 
101 ! midside nodes
102 
103  shape2d(5,i,j)=l2xi*l1eta
104  shape2d(6,i,j)=l3xi*l2eta
105  shape2d(7,i,j)=l2xi*l3eta
106  shape2d(8,i,j)=l1xi*l2eta
107 
108  dershape2d(1,5,i,j)=l2pxi*l1eta
109  dershape2d(1,6,i,j)=l3pxi*l2eta
110  dershape2d(1,7,i,j)=l2pxi*l3eta
111  dershape2d(1,8,i,j)=l1pxi*l2eta
112 
113  dershape2d(2,5,i,j)=l2xi*l1peta
114  dershape2d(2,6,i,j)=l3xi*l2peta
115  dershape2d(2,7,i,j)=l2xi*l3peta
116  dershape2d(2,8,i,j)=l1xi*l2peta
117 
118 ! center node
119 
120  shape2d(9,i,j)=l2xi*l2eta
121 
122  dershape2d(1,9,i,j)=l2pxi*l2eta
123  dershape2d(2,9,i,j)=l2xi*l2peta
124 
125  enddo
126  enddo
127 
128 ! check the 2D shape functions
129  do i = 1,nglla
130  do j = 1,ngllb
131 
132  sumshape=zero
133 
134  sumdershapexi=zero
135  sumdershapeeta=zero
136 
137  do ia = 1,ngnod2d
138 
139  sumshape=sumshape+shape2d(ia,i,j)
140 
141  sumdershapexi=sumdershapexi+dershape2d(1,ia,i,j)
142  sumdershapeeta=sumdershapeeta+dershape2d(2,ia,i,j)
143 
144  enddo
145 
146 ! the sum of the shape functions should be 1
147  if (abs(sumshape-one) > tinyval) call exit_mpi(myrank,'Error in 2D shape functions')
148 
149 ! the sum of the derivatives of the shape functions should be 0
150  if (abs(sumdershapexi) > tinyval) &
151  call exit_mpi(myrank,'Error in xi derivatives of 2D shape function')
152 
153  if (abs(sumdershapeeta) > tinyval) &
154  call exit_mpi(myrank,'Error in eta derivatives of 2D shape function')
155 
156  enddo
157  enddo
158 
159  end subroutine get_shape2d
