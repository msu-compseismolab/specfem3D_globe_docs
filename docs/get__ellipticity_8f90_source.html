28  subroutine get_ellipticity(xelm,yelm,zelm,nspl,rspl,espl,espl2)
29 
30  use constants
31 
32  implicit none
33 
34  integer :: nspl
35  double precision :: xelm(NGNOD)
36  double precision :: yelm(NGNOD)
37  double precision :: zelm(NGNOD)
38  double precision :: rspl(NR),espl(NR),espl2(NR)
39 
40  ! local parameters
41  integer :: ia
42 
43  double precision :: ell
44  double precision :: r,theta,phi,factor
45  double precision :: x,y,z
46  double precision :: cost,p20
47 
48  do ia = 1,ngnod
49 
50  x = xelm(ia)
51  y = yelm(ia)
52  z = zelm(ia)
53 
54  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
55 
56  cost = dcos(theta)
57 ! this is the Legendre polynomial of degree two, P2(cos(theta)), see the discussion above eq (14.4) in Dahlen and Tromp (1998)
58  p20 = 0.5d0*(3.0d0*cost*cost-1.0d0)
59 
60  ! get ellipticity using spline evaluation
61  call spline_evaluation(rspl,espl,espl2,nspl,r,ell)
62 
63 ! this is eq (14.4) in Dahlen and Tromp (1998)
64  factor = one-(two/3.0d0)*ell*p20
65 
66  xelm(ia) = x*factor
67  yelm(ia) = y*factor
68  zelm(ia) = z*factor
69 
70  enddo
71 
72  end subroutine get_ellipticity
