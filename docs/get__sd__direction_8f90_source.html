29 subroutine get_gradient_steepest_iso()
30 
31 ! calculates gradient by steepest descent method
32 
33  use tomography_kernels_iso
34 
35  implicit none
36  ! local parameters
37  real(kind=CUSTOM_REAL):: r,max,depth_max
38  ! gradient vector norm ( v^T * v )
39  real(kind=CUSTOM_REAL) :: norm_bulk,norm_beta,norm_rho
40  real(kind=CUSTOM_REAL) :: norm_bulk_sum,norm_beta_sum,norm_rho_sum
41  real(kind=CUSTOM_REAL) :: minmax(4)
42  real(kind=CUSTOM_REAL) :: min_beta,min_rho,max_beta,max_rho,min_bulk,max_bulk
43  integer :: iglob
44  integer :: i,j,k,ispec,ier
45 
46  ! allocate arrays for storing gradient
47  ! isotropic arrays
48  allocate(model_dbulk(ngllx,nglly,ngllz,nspec), &
49  model_dbeta(ngllx,nglly,ngllz,nspec), &
50  model_drho(ngllx,nglly,ngllz,nspec),stat=ier)
51  if (ier /= 0) stop 'error allocating gradient arrays'
52 
53  ! initializes arrays
54  model_dbulk = 0.0_custom_real
55  model_dbeta = 0.0_custom_real
56  model_drho = 0.0_custom_real
57 
58  ! initializes kernel maximum
59  max = 0._custom_real
60 
61  ! gradient in negative direction for steepest descent
62  do ispec = 1, nspec
63  do k = 1, ngllz
64  do j = 1, nglly
65  do i = 1, ngllx
66 
67  ! for bulk
68  model_dbulk(i,j,k,ispec) = - kernel_bulk(i,j,k,ispec)
69 
70  ! for shear
71  model_dbeta(i,j,k,ispec) = - kernel_beta(i,j,k,ispec)
72 
73  ! for rho
74  model_drho(i,j,k,ispec) = - kernel_rho(i,j,k,ispec)
75 
76  ! determines maximum kernel beta value within given radius
77  if (use_depth_range_maximum) then
78  ! get radius of point
79  iglob = ibool(i,j,k,ispec)
80  r = sqrt( x(iglob)*x(iglob) + y(iglob)*y(iglob) + z(iglob)*z(iglob) )
81 
82  ! stores maximum kernel betav value in this depth slice, since betav is most likely dominating
83  if (r < r_top .and. r > r_bottom) then
84  ! shear kernel value
85  max_beta = abs( kernel_beta(i,j,k,ispec) )
86  if (max < max_beta) then
87  max = max_beta
88  depth_max = r
89  endif
90  endif
91  endif
92 
93  enddo
94  enddo
95  enddo
96  enddo
97 
98  ! stores model_dbulk, ... arrays
99  call write_gradient_iso()
100 
101  ! statistics
102  call min_all_cr(minval(model_dbulk),min_bulk)
103  call max_all_cr(maxval(model_dbulk),max_bulk)
104 
105  call min_all_cr(minval(model_dbeta),min_beta)
106  call max_all_cr(maxval(model_dbeta),max_beta)
107 
108  call min_all_cr(minval(model_drho),min_rho)
109  call max_all_cr(maxval(model_drho),max_rho)
110 
111  if (myrank == 0) then
112  print *,'initial gradient:'
113  print *,' a min/max : ',min_bulk,max_bulk
114  print *,' beta min/max: ',min_beta,max_beta
115  print *,' rho min/max : ',min_rho,max_rho
116  print *
117  endif
118 
119  ! statistics output
120  if (print_statistics_files .and. myrank == 0) then
121  open(iout,file=trim(output_statistics_dir)//'statistics_gradient_minmax',status='unknown',action='write')
122  write(iout,*) '#min_beta #max_beta #min_bulk #max_bulk #min_rho #max_rho'
123  write(iout,'(6e24.12)') min_beta, max_beta, min_bulk, max_bulk, min_rho, max_rho
124  close(iout)
125  endif
126 
127  ! determines maximum kernel betav value within given radius
128  if (use_depth_range_maximum) then
129  ! maximum of all processes stored in max_vsv
130  call max_all_cr(max,max_beta)
131  max = max_beta
132  depth_max = r_earth_km *( 1.0 - depth_max )
133  endif
134 
135  ! determines step length based on maximum gradient value (either shear or bulk)
136  if (myrank == 0) then
137 
138  ! determines maximum kernel betav value within given radius
139  if (use_depth_range_maximum) then
140  print *,' using depth maximum: '
141  print *,' between depths (top/bottom) : ',r_top,r_bottom
142  print *,' maximum kernel value : ',max
143  print *,' depth of maximum kernel value : ',depth_max
144  print *
145  else
146  ! maximum gradient values
147  minmax(1) = abs(min_beta)
148  minmax(2) = abs(max_beta)
149  minmax(3) = abs(min_bulk)
150  minmax(4) = abs(max_bulk)
151 
152  ! maximum value of all kernel maxima
153  max = maxval(minmax)
154  endif
155  print *,'step length:'
156  print *,' using kernel maximum: ',max
157 
158  ! checks maximum value
159  if (max < 1.e-25) stop 'Error maximum kernel value too small for update'
160 
161  ! chooses step length such that it becomes the desired, given step factor as inputted
162  step_length = step_fac/max
163 
164  print *,' step length value : ',step_length
165  print *
166  endif
167  call bcast_all_singlecr(step_length)
168 
169 
170  ! gradient length sqrt( v^T * v )
171  norm_bulk = sum( model_dbulk * model_dbulk )
172  norm_beta = sum( model_dbeta * model_dbeta )
173  norm_rho = sum( model_drho * model_drho )
174 
175  call sum_all_cr(norm_bulk,norm_bulk_sum)
176  call sum_all_cr(norm_beta,norm_beta_sum)
177  call sum_all_cr(norm_rho,norm_rho_sum)
178 
179  if (myrank == 0) then
180  norm_bulk = sqrt(norm_bulk_sum)
181  norm_beta = sqrt(norm_beta_sum)
182  norm_rho = sqrt(norm_rho_sum)
183 
184  print *,'norm model updates:'
185  print *,' a : ',norm_bulk
186  print *,' beta: ',norm_beta
187  print *,' rho : ',norm_rho
188  print *
189  endif
190 
191  ! statistics output
192  if (print_statistics_files .and. myrank == 0) then
193  open(iout,file=trim(output_statistics_dir)//'statistics_vs_vp_rho_sum',status='unknown',action='write')
194  write(iout,*) '#norm_beta #norm_bulk #norm_rho'
195  write(iout,'(3e24.12)') norm_beta, norm_bulk, norm_rho
196  close(iout)
197  endif
198 
199  ! multiply model updates by a subjective factor that will change the step
200  model_dbulk(:,:,:,:) = step_length * model_dbulk(:,:,:,:)
201  model_dbeta(:,:,:,:) = step_length * model_dbeta(:,:,:,:)
202  model_drho(:,:,:,:) = step_length * model_drho(:,:,:,:)
203 
204  ! statistics
205  call min_all_cr(minval(model_dbulk),min_bulk)
206  call max_all_cr(maxval(model_dbulk),max_bulk)
207 
208  call min_all_cr(minval(model_dbeta),min_beta)
209  call max_all_cr(maxval(model_dbeta),max_beta)
210 
211  call min_all_cr(minval(model_drho),min_rho)
212  call max_all_cr(maxval(model_drho),max_rho)
213 
214  if (myrank == 0) then
215  print *,'scaled gradient:'
216  print *,' a min/max : ',min_bulk,max_bulk
217  print *,' beta min/max: ',min_beta,max_beta
218  print *,' rho min/max : ',min_rho,max_rho
219  print *
220  endif
221  call synchronize_all()
222 
223  ! statistics output
224  if (print_statistics_files .and. myrank == 0) then
225  open(iout,file=trim(output_statistics_dir)//'statistics_scaled_gradient',status='unknown',action='write')
226  write(iout,*) '#min_beta #max_beta #min_bulk #max_bulk #min_rho #max_rho'
227  write(iout,'(6e24.12)') min_beta,max_beta,min_bulk,max_bulk,min_rho,max_rho
228  close(iout)
229  endif
230 
231 end subroutine get_gradient_steepest_iso
