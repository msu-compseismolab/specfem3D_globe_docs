37 ! Gauss-Lobatto-Legendre points of integration and weights
38  double precision, dimension(NGLLX) :: xigll,wxgll
39  double precision, dimension(NGLLY) :: yigll,wygll
40  double precision, dimension(NGLLZ) :: zigll,wzgll
41 
42 ! array with derivatives of Lagrange polynomials and precalculated products
43  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLX) :: hprime_xx,hprimewgll_xx
44  real(kind=CUSTOM_REAL), dimension(NGLLY,NGLLY) :: hprime_yy,hprimewgll_yy

53 ! function for calculating derivatives of Lagrange polynomials
54  double precision, external :: lagrange_deriv_GLL
55 
56  integer i,j,k,i1,i2,j1,j2,k1,k2
57 
58 ! $$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$$
59 
60 ! set up coordinates of the Gauss-Lobatto-Legendre points
61  call zwgljd(xigll,wxgll,ngllx,gaussalpha,gaussbeta)
62  call zwgljd(yigll,wygll,nglly,gaussalpha,gaussbeta)
63  call zwgljd(zigll,wzgll,ngllz,gaussalpha,gaussbeta)
64 
65 ! calculate derivatives of the Lagrange polynomials
66 ! and precalculate some products in double precision
67 ! hprime(i,j) = h'_j(xigll_i) by definition of the derivation matrix
68  do i1 = 1,ngllx
69  do i2 = 1,ngllx
70  hprime_xx(i2,i1) = real(lagrange_deriv_GLL(i1-1,i2-1,xigll,NGLLX), kind=custom_real)
71  hprimewgll_xx(i2,i1) = real(lagrange_deriv_GLL(i1-1,i2-1,xigll,NGLLX)*wxgll(i2), kind=custom_real)
72  enddo
73  enddo
74 
75  do j1 = 1,nglly

77  hprime_yy(j2,j1) = real(lagrange_deriv_GLL(j1-1,j2-1,yigll,NGLLY), kind=custom_real)
78  hprimewgll_yy(j2,j1) = real(lagrange_deriv_GLL(j1-1,j2-1,yigll,NGLLY)*wygll(j2), kind=custom_real)
79  enddo
80  enddo
81 
82  do k1 = 1,ngllz
83  do k2 = 1,ngllz
84  hprime_zz(k2,k1) = real(lagrange_deriv_GLL(k1-1,k2-1,zigll,NGLLZ), kind=custom_real)
85  hprimewgll_zz(k2,k1) = real(lagrange_deriv_GLL(k1-1,k2-1,zigll,NGLLZ)*wzgll(k2), kind=custom_real)
86  enddo
87  enddo
88 
89  do i = 1,ngllx
90  do j = 1,nglly
91  wgllwgll_xy(i,j) = real(wxgll(i)*wygll(j), kind=custom_real)
92  enddo
93  enddo
94 
95  do i = 1,ngllx
96  do k = 1,ngllz
97  wgllwgll_xz(i,k) = real(wxgll(i)*wzgll(k), kind=custom_real)
98  enddo
99  enddo
100 
101  do j = 1,nglly
102  do k = 1,ngllz
103  wgllwgll_yz(j,k) = real(wygll(j)*wzgll(k), kind=custom_real)
104  enddo
105  enddo
106 
107  do i = 1,ngllx
108  do j = 1,nglly
109  do k = 1,ngllz
110  wgll_cube(i,j,k) = wxgll(i)*wygll(j)*wzgll(k)
111  enddo
112  enddo
113  enddo
