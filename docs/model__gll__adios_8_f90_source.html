90  else

96  call adios_schedule_read(file_handle_adios, sel, "reg1/vph/array", 0, 1, &
97  mgll_v%vph_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
98 
99  ! vs mesh
100  call adios_schedule_read(file_handle_adios, sel, "reg1/vsv/array", 0, 1, &
101  mgll_v%vsv_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
102  call adios_schedule_read(file_handle_adios, sel, "reg1/vsh/array", 0, 1, &
103  mgll_v%vsh_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
104 
105  ! eta mesh
106  call adios_schedule_read(file_handle_adios, sel, "reg1/eta/array", 0, 1, &
107  mgll_v%eta_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
108  endif
109 
110  ! rho mesh
111  call adios_schedule_read(file_handle_adios, sel, "reg1/rho/array", 0, 1, &
112  mgll_v%rho_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
113 
114  call adios_perform_reads(file_handle_adios, adios_err)
115  call check_adios_err(myrank,adios_err)
116 
117  call adios_read_close(file_handle_adios, adios_err)
118  call check_adios_err(myrank,adios_err)
119 
120 end subroutine read_gll_model_adios
