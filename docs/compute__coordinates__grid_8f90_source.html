68 ! loop on all the nodes in this element
69  do ignod = 1,ngnod
70 
71  if (ilayer == number_of_mesh_layers .and. include_central_cube) then
72 ! case of the inner core
73  ratio_xi = ((iproc_xi + offset_x(ignod)/dble(nex_per_proc_xi))/dble(nproc_xi))
74  fact_xi = 2.d0*ratio_xi-1.d0
75

78 
79  fact_xi_ = tan((angular_width_xi_rad/2.d0) * fact_xi)
80  fact_eta_ = tan((angular_width_eta_rad/2.d0) * fact_eta)
81 
82 ! uncomment the following lines to have more regular surface mesh (better aspect ratio for each element)
83 ! uncomment the corresponding lines in the else condition of this if statement too.
84 ! note that the ratio bigger_edge_size/smaller_edge_size for the surface mesh is a bit higher (1.43 vs 1.41)
85 
86 ! fact_xi_= (3.d0*fact_xi+4.d0*fact_xi_)/7.d0
87 ! fact_eta_= (3.d0*fact_eta+4.d0*fact_eta_)/7.d0
88 
89  xi = pi_over_two*fact_xi
90  eta = pi_over_two*fact_eta
91 
92  gamma = one / sqrt(one + fact_xi_**2 + fact_eta_**2)
93  rgt = (r_top / r_earth)*gamma
94 
95 ! coordinates of the edge extremity on the central cube surface
96  x_bot = ((r_bottom / r_earth) / sqrt(3.d0))* fact_xi * (1 + cos(eta)*central_cube_inflate_factor / pi)
97  y_bot = ((r_bottom / r_earth) / sqrt(3.d0)) * fact_eta * (1 + cos(xi)*central_cube_inflate_factor / pi)
98  z_bot = ((r_bottom / r_earth) / sqrt(3.d0)) * (1 + (cos(xi) + cos(eta))*central_cube_inflate_factor / pi)
99 
100 ! coordinates of the edge extremity on the ICB
101  x_top = fact_xi_*rgt
102  y_top = fact_eta_*rgt
103  z_top = rgt
104 
105  rn = offset_z(ignod) / dble(ner)
106  x = x_top*rn + x_bot*(one-rn)
107  y = y_top*rn + y_bot*(one-rn)
108  z = z_top*rn + z_bot*(one-rn)
109 
110  select case (ichunk)
111  case (chunk_ab)
112  xelm(ignod) = -y
113  yelm(ignod) = x
114  zelm(ignod) = z
115  case (chunk_ab_antipode)
116  xelm(ignod) = -y

118  zelm(ignod) = -z

123  case (chunk_ac_antipode)
124  xelm(ignod) = -y
125  yelm(ignod) = z
126  zelm(ignod) = -x
127  case (chunk_bc)
128  xelm(ignod) = -z
129  yelm(ignod) = y
130  zelm(ignod) = x
131  case (chunk_bc_antipode)
132  xelm(ignod) = z
133  yelm(ignod) = -y
134  zelm(ignod) = x
135  case default
136  stop 'incorrect chunk number in compute_coord_main_mesh'
137  end select
138 
139  else
140 
141 ! uncomment the following lines to have more regular surface mesh (better aspect ratio for each element)
142 ! note that the ratio bigger_edge_size/smaller_edge_size for the surface mesh is a bit higher (1.43 vs 1.41)
143 ! ratio_xi = ((iproc_xi + offset_x(ignod)/dble(NEX_PER_PROC_XI))/dble(NPROC_XI))*tan(ANGULAR_WIDTH_XI_RAD/2.d0)
144 ! x_ = 2.d0*ratio_xi-tan(ANGULAR_WIDTH_XI_RAD/2.d0)
145 ! ratio_eta = ((iproc_eta + offset_y(ignod)/dble(NEX_PER_PROC_ETA))/dble(NPROC_ETA))*tan(ANGULAR_WIDTH_ETA_RAD/2.d0)
146 ! y_ = 2.d0*ratio_eta-tan(ANGULAR_WIDTH_ETA_RAD/2.d0)
147 
148  ratio_xi = ((iproc_xi + offset_x(ignod)/dble(nex_per_proc_xi))/dble(nproc_xi))
149  x = 2.d0*ratio_xi-1
150 
151  ratio_eta = ((iproc_eta + offset_y(ignod)/dble(nex_per_proc_eta))/dble(nproc_eta))
152  y = 2.d0*ratio_eta-1
153 
154  x = tan((angular_width_xi_rad/2.d0) * x)
155  y = tan((angular_width_eta_rad/2.d0) * y)
156 
157 ! uncomment the following lines to have more regular surface mesh (better aspect ratio for each element)
158 ! note that the ratio bigger_edge_size/smaller_edge_size for the surface mesh is a bit higher (1.43 vs 1.41)
159 ! x= (3.d0*x_+4.d0*x)/7.d0
160 ! y= (3.d0*y_+4.d0*y)/7.d0
161 
162  gamma = one / sqrt(one + x*x + y*y)
163 
164  rgt = (r_top / r_earth)*gamma
165  rgb = (r_bottom / r_earth)*gamma
166 
167  ! define the mesh points on the top and the bottom in the six regions of the cubed sphere
168  select case (ichunk)
169 
170  case (chunk_ab)
171 
172  x_top = -y*rgt
173  y_top = x*rgt
174  z_top = rgt
175 
176  x_bot = -y*rgb
177  y_bot = x*rgb
178  z_bot = rgb
179 
180  case (chunk_ab_antipode)
181 
182  x_top = -y*rgt
183  y_top = -x*rgt
184  z_top = -rgt
185 
186  x_bot = -y*rgb
187  y_bot = -x*rgb
188  z_bot = -rgb
189 
190  case (chunk_ac)

193  y_top = -rgt
194  z_top = x*rgt

197  y_bot = -rgb

207  y_bot = rgb

233  end select
234 
235  ! rotate the chunk to the right location if we do not mesh the full Earth
236  if (nchunks /= 6) then
237 
238  ! rotate bottom
239  vector_ori(1) = x_bot
240  vector_ori(2) = y_bot
241  vector_ori(3) = z_bot
242  do i = 1,ndim

255  vector_ori(3) = z_top

261  enddo

263  y_top = vector_rotated(2)
264  z_top = vector_rotated(3)
265 
266  endif

268  ! compute the position of the point
269  rn = offset_z(ignod) / dble(ner)
270  xelm(ignod) = x_top*rn + x_bot*(one-rn)
271  yelm(ignod) = y_top*rn + y_bot*(one-rn)

274  endif

283  subroutine compute_coord_central_cube(ix,iy,iz, &
284  xgrid_central_cube,ygrid_central_cube,zgrid_central_cube, &
285  iproc_xi,iproc_eta,NPROC_XI,NPROC_ETA,nx_central_cube,ny_central_cube,nz_central_cube,radius_cube)
286 
287  use constants
288 
289  implicit none
290 
291  integer :: ix,iy,iz,iproc_xi,iproc_eta,NPROC_XI,NPROC_ETA,nx_central_cube,ny_central_cube,nz_central_cube
292 
293  double precision :: xgrid_central_cube,ygrid_central_cube,zgrid_central_cube,radius_cube
294 
295 ! local variables
296  double precision :: ratio_x,ratio_y,ratio_z
297  double precision :: fact_x,fact_y,fact_z,xi,eta,gamma
298 
299 ! the slice extends to the entire cube along Z
300 ! but only to current block along X and Y
301  ratio_x = (dble(iproc_xi) + dble(ix)/dble(2*nx_central_cube)) / dble(nproc_xi)
302  ratio_y = (dble(iproc_eta) + dble(iy)/dble(2*ny_central_cube)) / dble(nproc_eta)
303  ratio_z = dble(iz)/dble(2*nz_central_cube)
304 
305  if (abs(ratio_x) > 1.001d0 .or. abs(ratio_y) > 1.001d0 .or. abs(ratio_z) > 1.001d0) stop 'wrong ratio in central cube'
306 
307 ! use a "flat" cubed sphere to create the central cube
308 
309 ! map ratio to [-1,1] and then map to real radius
310 ! then add deformation
311  fact_x = 2.d0*ratio_x-1.d0
312  fact_y = 2.d0*ratio_y-1.d0
313  fact_z = 2.d0*ratio_z-1.d0
314 
315  xi = pi_over_two*fact_x
316  eta = pi_over_two*fact_y
317  gamma = pi_over_two*fact_z
318 
319  xgrid_central_cube = radius_cube * fact_x * (1 + (cos(eta)+cos(gamma))*central_cube_inflate_factor / pi)
320  ygrid_central_cube = radius_cube * fact_y * (1 + (cos(xi)+cos(gamma))*central_cube_inflate_factor / pi)
321  zgrid_central_cube = radius_cube * fact_z * (1 + (cos(xi)+cos(eta))*central_cube_inflate_factor / pi)
322 
323  end subroutine compute_coord_central_cube
