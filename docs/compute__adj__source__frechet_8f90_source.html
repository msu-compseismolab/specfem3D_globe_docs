29 ! compute the integrated derivatives of source parameters (M_jk and X_s)
30 
31  subroutine compute_adj_source_frechet(displ_s,Mxx,Myy,Mzz,Mxy,Mxz,Myz, &
32  eps_s,eps_m_s,eps_m_l_s, &
33  hxir,hetar,hgammar,hpxir,hpetar,hpgammar, hprime_xx,hprime_yy,hprime_zz, &
34  xix,xiy,xiz,etax,etay,etaz,gammax,gammay,gammaz)
35 
36  use constants
37 
38  implicit none
39 
40  ! input
41  real(kind=CUSTOM_REAL) :: displ_s(NDIM,NGLLX,NGLLY,NGLLZ)
42  double precision :: Mxx, Myy, Mzz, Mxy, Mxz, Myz
43  ! output
44  real(kind=CUSTOM_REAL) :: eps_s(NDIM,NDIM), eps_m_s, eps_m_l_s(NDIM)
45 
46  ! auxiliary
47  double precision :: hxir(NGLLX), hetar(NGLLY), hgammar(NGLLZ), &
48  hpxir(NGLLX),hpetar(NGLLY),hpgammar(NGLLZ)
49 
50  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLX) :: hprime_xx
51  real(kind=CUSTOM_REAL), dimension(NGLLY,NGLLY) :: hprime_yy
52  real(kind=CUSTOM_REAL), dimension(NGLLZ,NGLLZ) :: hprime_zz
53 
54  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLY,NGLLZ) :: &
55  xix,xiy,xiz,etax,etay,etaz,gammax,gammay,gammaz
56 
57 ! local variables
58  real(kind=CUSTOM_REAL) :: tempx1l,tempx2l,tempx3l, tempy1l,tempy2l,tempy3l, &
59  tempz1l,tempz2l,tempz3l, hp1, hp2, hp3, &
60  xixl,xiyl,xizl,etaxl,etayl,etazl,gammaxl,gammayl,gammazl, &
61  duxdxl,duxdyl,duxdzl,duydxl,duydyl,duydzl,duzdxl,duzdyl,duzdzl, &
62  xix_s,xiy_s,xiz_s,etax_s,etay_s,etaz_s,gammax_s,gammay_s,gammaz_s, &
63  hlagrange_xi, hlagrange_eta, hlagrange_gamma, hlagrange
64 
65  real(kind=CUSTOM_REAL) :: eps(NDIM,NDIM), eps_array(NDIM,NDIM,NGLLX,NGLLY,NGLLZ), &
66  eps_m_array(NGLLX,NGLLY,NGLLZ)
67 
68  integer i,j,k,l
69 
70 
71 ! first compute the strain at all the GLL points of the source element
72  do k = 1, ngllz
73  do j = 1, nglly
74  do i = 1, ngllx
75 
76  tempx1l = 0._custom_real
77  tempx2l = 0._custom_real
78  tempx3l = 0._custom_real
79 
80  tempy1l = 0._custom_real
81  tempy2l = 0._custom_real
82  tempy3l = 0._custom_real
83 
84  tempz1l = 0._custom_real
85  tempz2l = 0._custom_real
86  tempz3l = 0._custom_real
87 
88  do l = 1,ngllx
89  hp1 = hprime_xx(i,l)
90  tempx1l = tempx1l + displ_s(1,l,j,k)*hp1
91  tempy1l = tempy1l + displ_s(2,l,j,k)*hp1
92  tempz1l = tempz1l + displ_s(3,l,j,k)*hp1
93 
94  hp2 = hprime_yy(j,l)
95  tempx2l = tempx2l + displ_s(1,i,l,k)*hp2
96  tempy2l = tempy2l + displ_s(2,i,l,k)*hp2
97  tempz2l = tempz2l + displ_s(3,i,l,k)*hp2
98 
99  hp3 = hprime_zz(k,l)
100  tempx3l = tempx3l + displ_s(1,i,j,l)*hp3
101  tempy3l = tempy3l + displ_s(2,i,j,l)*hp3
102  tempz3l = tempz3l + displ_s(3,i,j,l)*hp3
103  enddo
104 
105 ! dudx
106  xixl = xix(i,j,k)
107  xiyl = xiy(i,j,k)
108  xizl = xiz(i,j,k)
109  etaxl = etax(i,j,k)
110  etayl = etay(i,j,k)
111  etazl = etaz(i,j,k)
112  gammaxl = gammax(i,j,k)
113  gammayl = gammay(i,j,k)
114  gammazl = gammaz(i,j,k)
115 
116  duxdxl = xixl*tempx1l + etaxl*tempx2l + gammaxl*tempx3l
117  duxdyl = xiyl*tempx1l + etayl*tempx2l + gammayl*tempx3l
118  duxdzl = xizl*tempx1l + etazl*tempx2l + gammazl*tempx3l
119 
120  duydxl = xixl*tempy1l + etaxl*tempy2l + gammaxl*tempy3l
121  duydyl = xiyl*tempy1l + etayl*tempy2l + gammayl*tempy3l
122  duydzl = xizl*tempy1l + etazl*tempy2l + gammazl*tempy3l
123 
124  duzdxl = xixl*tempz1l + etaxl*tempz2l + gammaxl*tempz3l
125  duzdyl = xiyl*tempz1l + etayl*tempz2l + gammayl*tempz3l
126  duzdzl = xizl*tempz1l + etazl*tempz2l + gammazl*tempz3l
127 
128 ! strain eps_jk
129  eps(1,1) = duxdxl
130  eps(1,2) = (duxdyl + duydxl) / 2
131  eps(1,3) = (duxdzl + duzdxl) / 2
132  eps(2,2) = duydyl
133  eps(2,3) = (duydzl + duzdyl) / 2
134  eps(3,3) = duzdzl
135  eps(2,1) = eps(1,2)
136  eps(3,1) = eps(1,3)
137  eps(3,2) = eps(2,3)
138 
139  eps_array(:,:,i,j,k) = eps(:,:)
140 
141 ! Mjk eps_jk
142  eps_m_array(i,j,k) = mxx * eps(1,1) + myy * eps(2,2) + mzz * eps(3,3) + &
143  2 * (mxy * eps(1,2) + mxz * eps(1,3) + myz * eps(2,3))
144 
145  enddo
146  enddo
147  enddo
148 
149  ! interpolate the strain eps_s(:,:) from eps_array(:,:,i,j,k)
150  eps_s = 0.; eps_m_s = 0.;
151  xix_s = 0.; xiy_s = 0.; xiz_s = 0.
152  etax_s = 0.; etay_s = 0.; etaz_s = 0.
153  gammax_s = 0.; gammay_s = 0.; gammaz_s = 0.
154 
155  do k = 1,ngllz
156  do j = 1,nglly
157  do i = 1,ngllx
158 
159  hlagrange = hxir(i)*hetar(j)*hgammar(k)
160 
161  eps_s(1,1) = eps_s(1,1) + eps_array(1,1,i,j,k)*hlagrange
162  eps_s(1,2) = eps_s(1,2) + eps_array(1,2,i,j,k)*hlagrange
163  eps_s(1,3) = eps_s(1,3) + eps_array(1,3,i,j,k)*hlagrange
164  eps_s(2,2) = eps_s(2,2) + eps_array(2,2,i,j,k)*hlagrange
165  eps_s(2,3) = eps_s(2,3) + eps_array(2,3,i,j,k)*hlagrange
166  eps_s(3,3) = eps_s(3,3) + eps_array(3,3,i,j,k)*hlagrange
167 
168  xix_s = xix_s + xix(i,j,k)*hlagrange
169  xiy_s = xiy_s + xiy(i,j,k)*hlagrange
170  xiz_s = xiz_s + xiz(i,j,k)*hlagrange
171  etax_s = etax_s + etax(i,j,k)*hlagrange
172  etay_s = etay_s + etay(i,j,k)*hlagrange
173  etaz_s = etaz_s + etaz(i,j,k)*hlagrange
174  gammax_s = gammax_s + gammax(i,j,k)*hlagrange
175  gammay_s = gammay_s + gammay(i,j,k)*hlagrange
176  gammaz_s = gammaz_s + gammaz(i,j,k)*hlagrange
177 
178  eps_m_s = eps_m_s + eps_m_array(i,j,k)*hlagrange
179  enddo
180  enddo
181  enddo
182 
183 ! for completion purpose, not used in specfem3D.f90
184  eps_s(2,1) = eps_s(1,2)
185  eps_s(3,1) = eps_s(1,3)
186  eps_s(3,2) = eps_s(2,3)
187 
188 ! compute the gradient of M_jk * eps_jk, and then interpolate it
189 
190  eps_m_l_s = 0.
191  do k = 1,ngllz
192  do j = 1,nglly
193  do i = 1,ngllx
194 
195  hlagrange_xi = hpxir(i)*hetar(j)*hgammar(k)
196  hlagrange_eta = hxir(i)*hpetar(j)*hgammar(k)
197  hlagrange_gamma = hxir(i)*hetar(j)*hpgammar(k)
198 
199  eps_m_l_s(1) = eps_m_l_s(1) + eps_m_array(i,j,k) * (hlagrange_xi * xix_s &
200  + hlagrange_eta * etax_s + hlagrange_gamma * gammax_s)
201  eps_m_l_s(2) = eps_m_l_s(2) + eps_m_array(i,j,k) * (hlagrange_xi * xiy_s &
202  + hlagrange_eta * etay_s + hlagrange_gamma * gammay_s)
203  eps_m_l_s(3) = eps_m_l_s(3) + eps_m_array(i,j,k) * (hlagrange_xi * xiz_s &
204  + hlagrange_eta * etaz_s + hlagrange_gamma * gammaz_s)
205 
206  enddo
207  enddo
208  enddo

210  end subroutine compute_adj_source_frechet
