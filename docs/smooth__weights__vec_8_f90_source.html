209  subroutine smooth_weights_cap_vardegree(lon,lat,xlon,xlat,weight,CAP_DEGREE,NTHETA,NPHI)
210 
211 ! calculates weighting points to smooth around lon/lat location with
212 ! a smoothing range of CAP_DEGREE
213 !
214 ! The cap is rotated to the North Pole.
215 !
216 ! returns: xlon,xlat,weight
217 
218  use constants, only: zero,one,tinyval,degrees_to_radians,radians_to_degrees,pi,two_pi,pi_over_two
219 
220  implicit none
221 
222  ! sampling rate
223  integer :: NTHETA
224  integer :: NPHI
225 
226  ! smoothing size (in degrees)
227  double precision :: CAP_DEGREE
228 
229  ! argument variables
230  double precision :: lat,lon
231  double precision,dimension(NTHETA*NPHI) :: xlon,xlat,weight
232 
233  ! local variables
234  double precision :: CAP
235  double precision :: theta,phi,sint,cost,sinp,cosp,wght,total
236  double precision :: r_rot,theta_rot,phi_rot
237  double precision :: rotation_matrix(3,3),x(3),xc(3)
238  double precision :: dtheta,dphi,cap_area,dweight,pi_over_nphi
239  integer :: i,j,k,itheta,iphi
240 
241  ! initializes
242  xlon(:) = zero
243  xlat(:) = zero
244  weight(:) = zero
245 
246  ! checks cap degree size
247  if (cap_degree < tinyval) then
248  ! no cap smoothing
249  print *,'Error cap:',cap_degree
250  print *,' lat/lon:',lat,lon
251  stop 'Error cap_degree too small'
252  endif
253 
254  ! pre-compute parameters
255  cap = cap_degree * degrees_to_radians
256  dtheta = 0.5d0 * cap / dble(ntheta)
257  dphi = two_pi / dble(nphi)
258 
259  cap_area = two_pi * ( one - dcos(cap) )
260  dweight = cap / dble(ntheta) * dphi / cap_area
261  pi_over_nphi = pi/dble(nphi)
262 
263  ! colatitude/longitude in radian
264  theta = (90.0d0 - lat ) * degrees_to_radians
265  phi = lon * degrees_to_radians
266 
267  sint = dsin(theta)
268  cost = dcos(theta)
269  sinp = dsin(phi)
270  cosp = dcos(phi)
271 
272  ! set up rotation matrix to go from cap at North pole to cap around point of interest
273  rotation_matrix(1,1) = cosp*cost
274  rotation_matrix(1,2) = -sinp
275  rotation_matrix(1,3) = cosp*sint
276  rotation_matrix(2,1) = sinp*cost
277  rotation_matrix(2,2) = cosp
278  rotation_matrix(2,3) = sinp*sint
279  rotation_matrix(3,1) = -sint
280  rotation_matrix(3,2) = zero
281  rotation_matrix(3,3) = cost
282 
283  ! calculates points over a cap at the North pole and rotates them to specified lat/lon point
284  i = 0
285  total = zero
286  do itheta = 1,ntheta
287 
288  theta = dble(2*itheta-1)*dtheta
289  cost = dcos(theta)
290  sint = dsin(theta)
291  wght = sint*dweight
292 
293  do iphi = 1,nphi
294 
295  i = i+1
296 
297  ! get the weight associated with this integration point (same for all phi)
298  weight(i) = wght
299 
300  total = total + weight(i)
301  phi = dble(2*iphi-1)*pi_over_nphi
302  cosp = dcos(phi)
303  sinp = dsin(phi)
304 
305  ! x,y,z coordinates of integration point in cap at North pole
306  xc(1) = sint*cosp
307  xc(2) = sint*sinp
308  xc(3) = cost
309 
310  ! get x,y,z coordinates in cap around point of interest
311  do j=1,3
312  x(j) = zero
313  do k=1,3
314  x(j) = x(j)+rotation_matrix(j,k)*xc(k)
315  enddo
316  enddo
317 
318  ! get latitude and longitude (degrees) of integration point
319  call xyz_2_rthetaphi_dble(x(1),x(2),x(3),r_rot,theta_rot,phi_rot)
320  call reduce(theta_rot,phi_rot)
321  xlat(i) = (pi_over_two - theta_rot) * radians_to_degrees
322  xlon(i) = phi_rot * radians_to_degrees
323  if (xlon(i) > 180.0d0) xlon(i) = xlon(i) - 360.0d0
324 
325  enddo
326 
327  enddo
328  if (abs(total - one) > 0.001d0) then
329  print *,'Error cap:',total,cap_degree
330  stop 'Error in cap integration for variable degree'
331  endif
332 
333  end subroutine smooth_weights_cap_vardegree
