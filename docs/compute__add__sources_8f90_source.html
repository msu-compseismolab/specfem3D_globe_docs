187  subroutine compute_add_sources_adjoint()
188 
189  use specfem_par
190  use specfem_par_crustmantle, only: accel_crust_mantle,ibool_crust_mantle
191 
192  implicit none
193 
194  ! local parameters
195  integer :: irec,irec_local,i,j,k,iglob
196  integer :: ivec_index
197  logical :: ibool_read_adj_arrays
198 
199  ! note: we check if nadj_rec_local > 0 before calling this routine, but better be safe...
200  if (nadj_rec_local == 0) return
201 
202  ! figure out if we need to read in a chunk of the adjoint source at this timestep
203  ibool_read_adj_arrays = ( (it == it_begin) .or. (mod(it-1,ntstep_between_read_adjsrc) == 0) )
204 
205  ! needs to read in a new chunk/block of the adjoint source
206  if (ibool_read_adj_arrays) then
207  call read_adjoint_sources()
208  endif
209 
210  ! adds adjoint sources
211  if (.not. gpu_mode) then
212  ! on CPU
213  irec_local = 0
214  do irec = 1,nrec
215 
216  ! adds source (only if this proc carries the source)
217  if (myrank == islice_selected_rec(irec)) then
218  irec_local = irec_local + 1
219 
220  ! adjoint source array index
221  ivec_index = iadj_vec(it)
222 
223  ! adds source contributions
224  do k = 1,ngllz
225  do j = 1,nglly
226  do i = 1,ngllx
227  iglob = ibool_crust_mantle(i,j,k,ispec_selected_rec(irec))
228 
229  ! adds adjoint source acting at this time step (it):
230  !
231  ! note: we use index iadj_vec(it) which is the corresponding time step
232  ! for the adjoint source acting at this time step (it)
233  !
234  ! see routine: setup_sources_receivers_adjindx() how this adjoint index array is set up
235  !
236  ! e.g. total length NSTEP = 3000, chunk length NTSTEP_BETWEEN_READ_ADJSRC= 1000
237  ! then for it = 1,..1000, first block has iadjsrc(1,1) with start = 2001 and end = 3000;
238  ! corresponding iadj_vec(it) goes from
239  ! iadj_vec(1) = 1000, iadj_vec(2) = 999 to iadj_vec(1000) = 1,
240  ! that is, originally the idea was
241  ! source_adjoint(.. iadj_vec(1) ) corresponds to adjoint source trace at time index 3000
242  ! source_adjoint(.. iadj_vec(2) ) corresponds to adjoint source trace at time index 2999
243  ! ..
244  ! source_adjoint(.. iadj_vec(1000) ) corresponds to adjoint source trace at time index 2001
245  ! then a new block will be read, etc, and it is going down till to adjoint source trace at time index 1
246  !
247  ! now comes the tricky part:
248  ! adjoint source traces are based on the seismograms from the forward run;
249  ! such seismograms have a time step index 1 which corresponds to time -t0
250  ! then time step index 2 which corresponds to -t0 + DT, and
251  ! the last time step in the file at time step NSTEP corresponds to time -t0 + (NSTEP-1)*DT
252  ! (see how we add the sources to the simulation in compute_add_sources() and
253  ! how we write/save the seismograms and wavefields at the end of the time loop).
254  !
255  ! then you use that seismogram and take e.g. the velocity of it for a traveltime adjoint source
256  !
257  ! now we read it in again, and remember the last time step in
258  ! the file at NSTEP corresponds to -t0 + (NSTEP-1)*DT
259  !
260  ! the same time step is saved for the forward wavefields to reconstruct them;
261  ! however, the Newmark time scheme acts at the very beginning of this time loop
262  ! such that we have the backward/reconstructed wavefield updated by
263  ! a single time step into the direction -DT and b_displ(it=1) would corresponds to -t0 + (NSTEP-1)*DT - DT
264  ! after the Newmark (predictor) time step update.
265  ! however, we will read the backward/reconstructed wavefield at the end of the first time loop,
266  ! such that b_displ(it=1) corresponds to -t0 + (NSTEP-1)*DT (which is the one saved in the files).
267  !
268  ! for the kernel calculations, we want:
269  ! adjoint wavefield at time t, starting from 0 to T
270  ! and forward wavefield at time T-t, starting from T down to 0
271  ! let's say time 0 corresponds to -t0 = -t0 + (it - 1)*DT at it=1
272  ! and time T corresponds to -t0 + (NSTEP-1)*DT at it = NSTEP
273  !
274  ! as seen before, the time for the forward wavefield b_displ(it=1) would then
275  ! correspond to time -t0 + (NSTEP-1)*DT - DT, which is T - DT.
276  ! the corresponding time for the adjoint wavefield thus would be 0 + DT
277  ! and the adjoint source index would be iadj_vec(it+1)
278  ! however, iadj_vec(it+1) which would go from 999 down to 0. 0 is out of bounds.
279  ! we thus would have to read in the adjoint source trace beginning from 2999 down to 0.
280  ! index 0 is not defined in the adjoint source trace, and would be set to zero.
281  !
282  ! however, since this complicates things, we read the backward/reconstructed
283  ! wavefield at the end of the first time loop, such that b_displ(it=1) corresponds to -t0 + (NSTEP-1)*DT.
284  ! assuming that until that end the backward/reconstructed wavefield and adjoint fields
285  ! have a zero contribution to adjoint kernels.
286  accel_crust_mantle(:,iglob) = accel_crust_mantle(:,iglob) &
287  + source_adjoint(:,irec_local,ivec_index)*(hxir_store(irec_local,i)*&
288  hetar_store(irec_local,j)*hgammar_store(irec_local,k))
289  enddo
290  enddo
291  enddo
292  endif
293 
294  enddo
295 
296  else
297 
298  ! on GPU
299  ! note: adjoint sourcearrays can become very big when used with many receiver stations
300  ! we overlap here the memory transfer to GPUs
301 
302  ! current time index
303  ivec_index = iadj_vec(it)
304 
305  if (gpu_async_copy) then
306  ! only synchronously transfers array at beginning or whenever new arrays were read in
307  if (ibool_read_adj_arrays) then
308  ! transfers adjoint arrays to GPU device memory
309  ! note: function call passes pointer to array source_adjoint at corresponding time slice
310  call transfer_adj_to_device(mesh_pointer,nrec,source_adjoint(1,1,ivec_index), &
311  islice_selected_rec)
312  endif
313  else
314  ! synchronously transfers adjoint arrays to GPU device memory before adding adjoint sources on GPU
315  call transfer_adj_to_device(mesh_pointer,nrec,source_adjoint(1,1,ivec_index), &
316  islice_selected_rec)
317  endif
318 
319  ! adds adjoint source contributions
320  call compute_add_sources_adjoint_gpu(mesh_pointer,nrec)
321 
322  if (gpu_async_copy) then
323  ! starts asynchronously transfer of next adjoint arrays to GPU device memory
324  ! (making sure the next source_adjoint values were already read in)
325  if ((.not. ibool_read_adj_arrays) .and. &
326  (.not. mod(it,ntstep_between_read_adjsrc) == 0) .and. &
327  (.not. it == it_end)) then
328  ! next time index
329  ivec_index = iadj_vec(it+1)
330 
331  ! checks next index
332  if (ivec_index < 1 .or. ivec_index > ntstep_between_read_adjsrc) then
333  print *,'Error iadj_vec bounds: rank',myrank,' it = ',it,' index = ',ivec_index, &
334  'out of bounds ',1,'to',ntstep_between_read_adjsrc
335  call exit_mpi(myrank,'Error iadj_vec index bounds')
336  endif
337 
338  ! asynchronously transfers next time slice
339  call transfer_adj_to_device_async(mesh_pointer,nrec,source_adjoint(1,1,ivec_index), &
340  islice_selected_rec)
341  endif
342  endif
343 
344  endif
345 
346  end subroutine compute_add_sources_adjoint
