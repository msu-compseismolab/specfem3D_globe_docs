30  subroutine prepare_attenuation()
31 
32  ! precomputes attenuation factors
33 
34  use specfem_par
35  use specfem_par_crustmantle
36  use specfem_par_innercore
37  use specfem_par_movie, only: muvstore_crust_mantle_3dmovie
38 
39  implicit none
40 
41  ! local parameters
42  double precision, dimension(N_SLS) :: alphaval_dble, betaval_dble, gammaval_dble
43 
44  double precision :: scale_factor,scale_factor_minus_one
45  real(kind=CUSTOM_REAL) :: mul
46  integer :: ispec,i,j,k,ier
47 
48  ! checks if attenuation is on and anything to do
49  if (.not. attenuation_val ) return
50 
51  ! get and store PREM attenuation model
52  if (myrank == 0) then
53  write(imain,*) "preparing attenuation"
54  call flush_imain()
55  endif
56 
57  ! allocates arrays
58  allocate(one_minus_sum_beta_crust_mantle(att1_val,att2_val,att3_val,att4_val), &
59  factor_scale_crust_mantle(att1_val,att2_val,att3_val,att4_val), &
60  factor_common_crust_mantle(att1_val,att2_val,att3_val,n_sls,att4_val),stat=ier)
61  if (ier /= 0) stop 'Error allocating arrays one_minus_sum_beta_crust_mantle,..'
62 
63  allocate(one_minus_sum_beta_inner_core(att1_val,att2_val,att3_val,att5_val), &
64  factor_scale_inner_core(att1_val,att2_val,att3_val,att5_val), &
65  factor_common_inner_core(att1_val,att2_val,att3_val,n_sls,att5_val),stat=ier)
66  if (ier /= 0) stop 'Error allocating arrays one_minus_sum_beta_inner_core,..'
67 
68  ! reads in attenuation values
69  ! CRUST_MANTLE ATTENUATION
70  call get_attenuation_model_3d(iregion_crust_mantle, &
71  one_minus_sum_beta_crust_mantle, &
72  factor_common_crust_mantle, &
73  factor_scale_crust_mantle,tau_sigma_dble, &
74  nspec_crust_mantle)
75  call bcast_attenuation_model_3d(one_minus_sum_beta_crust_mantle, &
76  factor_common_crust_mantle, &
77  factor_scale_crust_mantle, &
78  tau_sigma_dble, &
79  nspec_crust_mantle)
80 
81  ! INNER_CORE ATTENUATION
82  call get_attenuation_model_3d(iregion_inner_core, &
83  one_minus_sum_beta_inner_core, &
84  factor_common_inner_core, &
85  factor_scale_inner_core,tau_sigma_dble, &
86  nspec_inner_core)
87  call bcast_attenuation_model_3d(one_minus_sum_beta_inner_core, &
88  factor_common_inner_core, &
89  factor_scale_inner_core, &
90  tau_sigma_dble, &
91  nspec_inner_core)
92 
93  ! physical dispersion: scales moduli from reference frequency to simulation (source) center frequency
94  !
95  ! if attenuation is on, shift PREM to right frequency
96  ! rescale mu in PREM to average frequency for attenuation
97  !
98  ! the formulas to implement the scaling can be found for instance in
99  ! Liu, H. P., Anderson, D. L. and Kanamori, H., Velocity dispersion due to
100  ! anelasticity: implications for seismology and mantle composition,
101  ! Geophys. J. R. Astron. Soc., vol. 47, pp. 41-58 (1976)
102  !
103  ! and in Aki, K. and Richards, P. G., Quantitative seismology, theory and methods,
104  ! W. H. Freeman, (1980), second edition, sections 5.5 and 5.5.2, eq. (5.81) p. 170.
105  !
106  ! Beware that in the book of Aki and Richards eq. (5.81) is given for velocities
107  ! while we need an equation for "mu" and thus we have an additional factor of 2
108  ! in the scaling factor below and in equation (49) of Komatitsch and Tromp, Geophys. J. Int. (2002) 149, 390-412,
109  ! because "mu" is related to the square of velocity.
110 
111  ! rescale in crust and mantle
112  do ispec = 1,nspec_crust_mantle
113  do k = 1,ngllz
114  do j = 1,nglly
115  do i = 1,ngllx
116 
117  if (attenuation_3d_val .or. attenuation_1d_with_3d_storage_val) then
118  scale_factor = factor_scale_crust_mantle(i,j,k,ispec)
119  else
120  scale_factor = factor_scale_crust_mantle(1,1,1,ispec)
121  endif
122 
123  if (anisotropic_3d_mantle_val) then
124  scale_factor_minus_one = scale_factor - 1.d0
125 
126  mul = c44store_crust_mantle(i,j,k,ispec)
127  c11store_crust_mantle(i,j,k,ispec) = c11store_crust_mantle(i,j,k,ispec) &
128  + four_thirds * scale_factor_minus_one * mul
129  c12store_crust_mantle(i,j,k,ispec) = c12store_crust_mantle(i,j,k,ispec) &
130  - two_thirds * scale_factor_minus_one * mul
131  c13store_crust_mantle(i,j,k,ispec) = c13store_crust_mantle(i,j,k,ispec) &
132  - two_thirds * scale_factor_minus_one * mul
133  c22store_crust_mantle(i,j,k,ispec) = c22store_crust_mantle(i,j,k,ispec) &
134  + four_thirds * scale_factor_minus_one * mul
135  c23store_crust_mantle(i,j,k,ispec) = c23store_crust_mantle(i,j,k,ispec) &
136  - two_thirds * scale_factor_minus_one * mul
137  c33store_crust_mantle(i,j,k,ispec) = c33store_crust_mantle(i,j,k,ispec) &
138  + four_thirds * scale_factor_minus_one * mul
139  c44store_crust_mantle(i,j,k,ispec) = c44store_crust_mantle(i,j,k,ispec) &
140  + scale_factor_minus_one * mul
141  c55store_crust_mantle(i,j,k,ispec) = c55store_crust_mantle(i,j,k,ispec) &
142  + scale_factor_minus_one * mul
143  c66store_crust_mantle(i,j,k,ispec) = c66store_crust_mantle(i,j,k,ispec) &
144  + scale_factor_minus_one * mul
145  else
146  if (movie_volume .and. simulation_type == 3) then
147  ! store the original value of \mu to compute \mu*\eps
148  muvstore_crust_mantle_3dmovie(i,j,k,ispec) = muvstore_crust_mantle(i,j,k,ispec)
149  endif
150 
151  muvstore_crust_mantle(i,j,k,ispec) = muvstore_crust_mantle(i,j,k,ispec) * scale_factor
152 
153  ! scales transverse isotropic values for mu_h
154  if (ispec_is_tiso_crust_mantle(ispec)) then
155  muhstore_crust_mantle(i,j,k,ispec) = muhstore_crust_mantle(i,j,k,ispec) * scale_factor
156  endif
157  endif
158 
159  enddo
160  enddo
161  enddo
162  enddo ! enddo CRUST MANTLE
163 
164  ! rescale in inner core
165  do ispec = 1,nspec_inner_core
166  do k = 1,ngllz
167  do j = 1,nglly
168  do i = 1,ngllx
169 
170  if (attenuation_3d_val .or. attenuation_1d_with_3d_storage_val) then
171  scale_factor = factor_scale_inner_core(i,j,k,ispec)
172  else
173  scale_factor = factor_scale_inner_core(1,1,1,ispec)
174  endif
175 
176  if (anisotropic_inner_core_val) then
177  scale_factor_minus_one = scale_factor - 1.d0
178 
179  mul = c44store_inner_core(i,j,k,ispec)
180  c11store_inner_core(i,j,k,ispec) = c11store_inner_core(i,j,k,ispec) &
181  + four_thirds * scale_factor_minus_one * mul
182  c12store_inner_core(i,j,k,ispec) = c12store_inner_core(i,j,k,ispec) &
183  - two_thirds * scale_factor_minus_one * mul
184  c13store_inner_core(i,j,k,ispec) = c13store_inner_core(i,j,k,ispec) &
185  - two_thirds * scale_factor_minus_one * mul
186  c33store_inner_core(i,j,k,ispec) = c33store_inner_core(i,j,k,ispec) &
187  + four_thirds * scale_factor_minus_one * mul
188  c44store_inner_core(i,j,k,ispec) = c44store_inner_core(i,j,k,ispec) &
189  + scale_factor_minus_one * mul
190  endif
191 
192  muvstore_inner_core(i,j,k,ispec) = muvstore_inner_core(i,j,k,ispec) * scale_factor
193 
194  enddo
195  enddo
196  enddo
197  enddo ! enddo INNER CORE
198 
199  ! precompute Runge-Kutta coefficients
200  call get_attenuation_memory_values(tau_sigma_dble, deltat, alphaval_dble, betaval_dble, gammaval_dble)
201  alphaval = real(alphaval_dble, kind=custom_real)
202  betaval = real(betaval_dble, kind=custom_real)
203  gammaval = real(gammaval_dble, kind=custom_real)
204 
205  if (simulation_type == 3) then
206  call get_attenuation_memory_values(tau_sigma_dble, b_deltat, alphaval_dble, betaval_dble, gammaval_dble)
207  b_alphaval = real(alphaval_dble, kind=custom_real)
208  b_betaval = real(betaval_dble, kind=custom_real)
209  b_gammaval = real(gammaval_dble, kind=custom_real)
210  endif
211 
212  if (use_lddrk) then
213  tau_sigma_custom_real(:) = real(tau_sigma_dble(:), kind=custom_real)
214  endif
215 
216  if (undo_attenuation) then
217  b_alphaval = alphaval
218  b_betaval = betaval
219  b_gammaval = gammaval
220  endif
221 
222  ! user output
223  if (myrank == 0) then
224  write(imain,*) " attenuation period range min/max: ",min_attenuation_period,'/',max_attenuation_period,' (s)'
225  write(imain,*) " ATTENUATION_1D_WITH_3D_STORAGE : ",attenuation_1d_with_3d_storage_val
226  write(imain,*) " ATTENUATION_3D : ",attenuation_3d_val
227  call flush_imain()
228  endif
229 
230  ! synchronizes processes
231  call synchronize_all()
232 
233  end subroutine prepare_attenuation
